 65 function _rsasign_getHexPaddedDigestInfoForString(s, keySize, hashAlg) {
 66     var pmStrLen = keySize / 4;
 67     var hashFunc = function(s) { return KJUR.crypto.Util.hashString(s, hashAlg); };
 68     var sHashHex = hashFunc(s);
 69 
 70     var sHead = "0001";
 71     var sTail = "00" + _RSASIGN_DIHEAD[hashAlg] + sHashHex;
 72     var sMid = "";
 73     var fLen = pmStrLen - sHead.length - sTail.length;
 74     for (var i = 0; i < fLen; i += 2) {
 75 	sMid += "ff";
 76     }
 77     sPaddedMessageHex = sHead + sMid + sTail;
 78     return sPaddedMessageHex;
 79 }

 81 function _zeroPaddingOfSignature(hex, bitLength) {
 82     var s = "";
 83     var nZero = bitLength / 4 - hex.length;
 84     for (var i = 0; i < nZero; i++) {
 85 	s = s + "0";
 86     }
 87     return s + hex;
 88 }

 90 /**
 91  * sign for a message string with RSA private key.<br/>
 92  * @name signString
 93  * @memberOf RSAKey#
 94  * @function
 95  * @param {String} s message string to be signed.
 96  * @param {String} hashAlg hash algorithm name for signing.<br/>
 97  * @return returns hexadecimal string of signature value.
 98  */
 99 function _rsasign_signString(s, hashAlg) {
100     //alert("this.n.bitLength() = " + this.n.bitLength());
101     var hPM = _rsasign_getHexPaddedDigestInfoForString(s, this.n.bitLength(), hashAlg);
102     var biPaddedMessage = parseBigInt(hPM, 16);
103     var biSign = this.doPrivate(biPaddedMessage);
104     var hexSign = biSign.toString(16);
105     return _zeroPaddingOfSignature(hexSign, this.n.bitLength());
106 }

108 function _rsasign_signStringWithSHA1(s) {
109     return _rsasign_signString.call(this, s, 'sha1');
110 }
111 
112 function _rsasign_signStringWithSHA256(s) {
113     return _rsasign_signString.call(this, s, 'sha256');

116 // PKCS#1 (PSS) mask generation function
117 function pss_mgf1_str(seed, len, hash) {
118     var mask = '', i = 0;
119 
120     while (mask.length < len) {
121         mask += hextorstr(hash(rstrtohex(seed + String.fromCharCode.apply(String, [
122                 (i & 0xff000000) >> 24,
123                 (i & 0x00ff0000) >> 16,
124                 (i & 0x0000ff00) >> 8,
125                 i & 0x000000ff]))));
126         i += 1;
127     }
128 
129     return mask;
130 }

132 /**
133  * sign for a message string with RSA private key by PKCS#1 PSS signing.<br/>
134  * @name signStringPSS
135  * @memberOf RSAKey
136  * @function
137  * @param {String} s message string to be signed.
138  * @param {String} hashAlg hash algorithm name for signing.
139  * @param {Integer} sLen salt length (-1 or -2)
140  * @return returns hexadecimal string of signature value.
141  */
142 function _rsasign_signStringPSS(s, hashAlg, sLen) {
143     var hashFunc = function(sHex) { return KJUR.crypto.Util.hashHex(sHex, hashAlg); } 
144     var hHash = hashFunc(rstrtohex(s));
145     var mHash = hextorstr(hHash);
146     var hLen = mHash.length;
147     var emBits = this.n.bitLength() - 1;
148     var emLen = Math.ceil(emBits / 8);
149     var i;
150 
151     if (sLen === -1) {
152         sLen = hLen; // same as hash length
153     } else if ((sLen === -2) || (sLen === undefined)) {
154         sLen = emLen - hLen - 2; // maximum
155     } else if (sLen < -2) {
156         throw "invalid salt length";
157     }
158 
159     if (emLen < (hLen + sLen + 2)) {
160         throw "data too long";
161     }
162 
163     var salt = '';
164 
165     if (sLen > 0) {
166         salt = new Array(sLen);
167         new SecureRandom().nextBytes(salt);
168         salt = String.fromCharCode.apply(String, salt);
169     }
170 
171     var H = hextorstr(hashFunc(rstrtohex('\x00\x00\x00\x00\x00\x00\x00\x00' + mHash + salt)));
172     var PS = [];
173 
174     for (i = 0; i < emLen - sLen - hLen - 2; i += 1) {
175         PS[i] = 0x00;
176     }
177 
178     var DB = String.fromCharCode.apply(String, PS) + '\x01' + salt;
179     var dbMask = pss_mgf1_str(H, DB.length, hashFunc);
180     var maskedDB = [];
181 
182     for (i = 0; i < DB.length; i += 1) {
183         maskedDB[i] = DB.charCodeAt(i) ^ dbMask.charCodeAt(i);
184     }
185 
186     var mask = (0xff00 >> (8 * emLen - emBits)) & 0xff;
187     maskedDB[0] &= ~mask;
188 
189     for (i = 0; i < hLen; i++) {
190         maskedDB.push(H.charCodeAt(i));
191     }
192 
193     maskedDB.push(0xbc);
194 
195     return _zeroPaddingOfSignature(this.doPrivate(new BigInteger(maskedDB)).toString(16),
196 				   this.n.bitLength());
197 }

203 function _rsasign_getDecryptSignatureBI(biSig, hN, hE) {
204     var rsa = new RSAKey();
205     rsa.setPublic(hN, hE);
206     var biDecryptedSig = rsa.doPublic(biSig);
207     return biDecryptedSig;
208 }

210 function _rsasign_getHexDigestInfoFromSig(biSig, hN, hE) {
211     var biDecryptedSig = _rsasign_getDecryptSignatureBI(biSig, hN, hE);
212     var hDigestInfo = biDecryptedSig.toString(16).replace(/^1f+00/, '');
213     return hDigestInfo;
214 }

216 function _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo) {
217     for (var algName in _RSASIGN_DIHEAD) {
218 	var head = _RSASIGN_DIHEAD[algName];
219 	var len = head.length;
220 	if (hDigestInfo.substring(0, len) == head) {
221 	    var a = [algName, hDigestInfo.substring(len)];
222 	    return a;
223 	}
224     }
225     return [];
226 }

228 function _rsasign_verifySignatureWithArgs(sMsg, biSig, hN, hE) {
229     var hDigestInfo = _rsasign_getHexDigestInfoFromSig(biSig, hN, hE);
230     var digestInfoAry = _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo);
231     if (digestInfoAry.length == 0) return false;
232     var algName = digestInfoAry[0];
233     var diHashValue = digestInfoAry[1];
234     var ff = function(s) { return KJUR.crypto.Util.hashString(s, algName); };
235     var msgHashValue = ff(sMsg);
236     return (diHashValue == msgHashValue);
237 }

239 function _rsasign_verifyHexSignatureForMessage(hSig, sMsg) {
240     var biSig = parseBigInt(hSig, 16);
241     var result = _rsasign_verifySignatureWithArgs(sMsg, biSig,
242 						  this.n.toString(16),
243 						  this.e.toString(16));
244     return result;

247 /**
248  * verifies a sigature for a message string with RSA public key.<br/>
249  * @name verifyString
250  * @memberOf RSAKey#
251  * @function
252  * @param {String} sMsg message string to be verified.
253  * @param {String} hSig hexadecimal string of siganture.<br/>
254  *                 non-hexadecimal charactors including new lines will be ignored.
255  * @return returns 1 if valid, otherwise 0
256  */
257 function _rsasign_verifyString(sMsg, hSig) {
258     hSig = hSig.replace(_RE_HEXDECONLY, '');
259     hSig = hSig.replace(/[ \n]+/g, "");
260     var biSig = parseBigInt(hSig, 16);
261     if (biSig.bitLength() > this.n.bitLength()) return 0;
262     var biDecryptedSig = this.doPublic(biSig);
263     var hDigestInfo = biDecryptedSig.toString(16).replace(/^1f+00/, '');
264     var digestInfoAry = _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo);
265   
266     if (digestInfoAry.length == 0) return false;
267     var algName = digestInfoAry[0];
268     var diHashValue = digestInfoAry[1];
269     var ff = function(s) { return KJUR.crypto.Util.hashString(s, algName); };
270     var msgHashValue = ff(sMsg);
271     return (diHashValue == msgHashValue);
272 }

274 /**
275  * verifies a sigature for a message string with RSA public key by PKCS#1 PSS sign.<br/>
276  * @name verifyStringPSS
277  * @memberOf RSAKey
278  * @function
279  * @param {String} sMsg message string to be verified.
280  * @param {String} hSig hexadecimal string of signature value
281  * @param {String} hashAlg hash algorithm name
282  * @param {Integer} sLen salt length for PSS signature (-1 or -2)
283  * @return returns true if valid, otherwise false
284  */
285 function _rsasign_verifyStringPSS(sMsg, hSig, hashAlg, sLen) {
286     var biSig = new BigInteger(hSig, 16);
287 
288     if (biSig.bitLength() > this.n.bitLength()) {
289         return false;
290     }
291 
292     var hashFunc = function(sHex) { return KJUR.crypto.Util.hashHex(sHex, hashAlg); };
293     var hHash = hashFunc(rstrtohex(sMsg));
294     var mHash = hextorstr(hHash);
295     var hLen = mHash.length;
296     var emBits = this.n.bitLength() - 1;
297     var emLen = Math.ceil(emBits / 8);
298     var i;
299 
300     if (sLen === -1) {
301         sLen = hLen; // same as hash length
302     } else if ((sLen === -2) || (sLen === undefined)) {
303         sLen = emLen - hLen - 2; // recover
304     } else if (sLen < -2) {
305         throw "invalid salt length";
306     }
307 
308     if (emLen < (hLen + sLen + 2)) {
309         throw "data too long";
310     }
311 
312     var em = this.doPublic(biSig).toByteArray();
313 
314     for (i = 0; i < em.length; i += 1) {
315         em[i] &= 0xff;
316     }
317 
318     while (em.length < emLen) {
319         em.unshift(0);
320     }
321 
322     if (em[emLen -1] !== 0xbc) {
323         throw "encoded message does not end in 0xbc";
324     }
325 
326     em = String.fromCharCode.apply(String, em);
327 
328     var maskedDB = em.substr(0, emLen - hLen - 1);
329     var H = em.substr(maskedDB.length, hLen);
330 
331     var mask = (0xff00 >> (8 * emLen - emBits)) & 0xff;
332 
333     if ((maskedDB.charCodeAt(0) & mask) !== 0) {
334         throw "bits beyond keysize not zero";
335     }
336 
337     var dbMask = pss_mgf1_str(H, maskedDB.length, hashFunc);
338     var DB = [];
339 
340     for (i = 0; i < maskedDB.length; i += 1) {
341         DB[i] = maskedDB.charCodeAt(i) ^ dbMask.charCodeAt(i);
342     }
343 
344     DB[0] &= ~mask;
345 
346     var checkLen = emLen - hLen - sLen - 2;
347 
348     for (i = 0; i < checkLen; i += 1) {
349         if (DB[i] !== 0x00) {
350             throw "leftmost octets not zero";
351         }
352     }
353 
354     if (DB[checkLen] !== 0x01) {
355         throw "0x01 marker not found";
356     }
357 
358     return H === hextorstr(hashFunc(rstrtohex('\x00\x00\x00\x00\x00\x00\x00\x00' + mHash +
359 				     String.fromCharCode.apply(String, DB.slice(-sLen)))));
360 }
