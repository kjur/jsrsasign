 41  * @property {Dictionary} _RSASIGN_DIHEAD
 42  * @description Array of head part of hexadecimal DigestInfo value for hash algorithms.
 43  * You can add any DigestInfo hash algorith for signing.
 44  * See PKCS#1 v2.1 spec (p38).

 47 _RSASIGN_DIHEAD['sha1'] =      "3021300906052b0e03021a05000414";
 48 _RSASIGN_DIHEAD['sha256'] =    "3031300d060960864801650304020105000420";

 55 /**
 56  * @property {Dictionary} _RSASIGN_HASHHEXFUNC
 57  * @description Array of functions which calculate hash and returns it as hexadecimal.
 58  * You can add any hash algorithm implementations.
 59  */
 60 var _RSASIGN_HASHHEXFUNC = [];
 61 _RSASIGN_HASHHEXFUNC['sha1'] =      function(s){return hex_sha1(s);};  // http://pajhome.org.uk/crypt/md5/md5.html
 62 _RSASIGN_HASHHEXFUNC['sha256'] =    function(s){return hex_sha256(s);} // http://pajhome.org.uk/crypt/md5/md5.html
 63 _RSASIGN_HASHHEXFUNC['sha512'] =    function(s){return hex_sha512(s);} // http://pajhome.org.uk/crypt/md5/md5.html
 64 _RSASIGN_HASHHEXFUNC['md5'] =       function(s){return hex_md5(s);};   // http://pajhome.org.uk/crypt/md5/md5.html

 77 function _rsasign_getHexPaddedDigestInfoForString(s, keySize, hashAlg) {
 78     var pmStrLen = keySize / 4;
 79     var hashFunc = _RSASIGN_HASHHEXFUNC[hashAlg];
 80     var sHashHex = hashFunc(s);
 81 
 82     var sHead = "0001";
 83     var sTail = "00" + _RSASIGN_DIHEAD[hashAlg] + sHashHex;
 84     var sMid = "";
 85     var fLen = pmStrLen - sHead.length - sTail.length;
 86     for (var i = 0; i < fLen; i += 2) {
 87 	sMid += "ff";
 88     }
 89     sPaddedMessageHex = sHead + sMid + sTail;
 90     return sPaddedMessageHex;
 91 }

 93 function _zeroPaddingOfSignature(hex, bitLength) {
 94     var s = "";
 95     var nZero = bitLength / 4 - hex.length;
 96     for (var i = 0; i < nZero; i++) {
 97 	s = s + "0";
 98     }
 99     return s + hex;
100 }

102 /**
103  * sign for a message string with RSA private key.<br/>
104  * @name signString
105  * @memberOf RSAKey#
106  * @function
107  * @param {String} s message string to be signed.
108  * @param {String} hashAlg hash algorithm name for signing.<br/>
109  * @return returns hexadecimal string of signature value.
110  */
111 function _rsasign_signString(s, hashAlg) {
112     //alert("this.n.bitLength() = " + this.n.bitLength());
113     var hPM = _rsasign_getHexPaddedDigestInfoForString(s, this.n.bitLength(), hashAlg);
114     var biPaddedMessage = parseBigInt(hPM, 16);
115     var biSign = this.doPrivate(biPaddedMessage);
116     var hexSign = biSign.toString(16);
117     return _zeroPaddingOfSignature(hexSign, this.n.bitLength());
118 }

120 function _rsasign_signStringWithSHA1(s) {
121     return _rsasign_signString.call(this, s, 'sha1');
122 }

124 function _rsasign_signStringWithSHA256(s) {
125     return _rsasign_signString.call(this, s, 'sha256');

128 // PKCS#1 (PSS) mask generation function
129 function pss_mgf1_str(seed, len, hash) {
130     var mask = '', i = 0;
131 
132     while (mask.length < len) {
133         mask += hash(seed + String.fromCharCode.apply(String, [
134                 (i & 0xff000000) >> 24,
135                 (i & 0x00ff0000) >> 16,
136                 (i & 0x0000ff00) >> 8,
137                 i & 0x000000ff]));
138         i += 1;
139     }
140 
141     return mask;
142 }

144 /**
145  * sign for a message string with RSA private key by PKCS#1 PSS signing.<br/>
146  * @name signStringPSS
147  * @memberOf RSAKey#
148  * @function
149  * @param {String} s message string to be signed.
150  * @param {String} hashAlg hash algorithm name for signing.<br/>
151  * @return returns hexadecimal string of signature value.
152  */
153 function _rsasign_signStringPSS(s, hashAlg, sLen) {
154     var hashFunc = _RSASIGN_HASHRAWFUNC[hashAlg];
155     var mHash = hashFunc(s);
156     var hLen = mHash.length;
157     var emBits = this.n.bitLength() - 1;
158     var emLen = Math.ceil(emBits / 8);
159     var i;
160 
161     if (sLen === -1) {
162         sLen = hLen; // same has hash length
163     } else if ((sLen === -2) || (sLen === undefined)) {
164         sLen = emLen - hLen - 2; // maximum
165     } else if (sLen < -2) {
166         throw "invalid salt length";
167     }
168 
169     if (emLen < (hLen + sLen + 2)) {
170         throw "data too long";
171     }
172 
173     var salt = '';
174 
175     if (sLen > 0) {
176         salt = new Array(sLen);
177         new SecureRandom().nextBytes(salt);
178         salt = String.fromCharCode.apply(String, salt);
179     }
180 
181     var H = hashFunc('\x00\x00\x00\x00\x00\x00\x00\x00' + mHash + salt);
182     var PS = [];
183 
184     for (i = 0; i < emLen - sLen - hLen - 2; i += 1) {
185         PS[i] = 0x00;
186     }
187 
188     var DB = String.fromCharCode.apply(String, PS) + '\x01' + salt;
189     var dbMask = pss_mgf1_str(H, DB.length, hashFunc);
190     var maskedDB = [];
191 
192     for (i = 0; i < DB.length; i += 1) {
193         maskedDB[i] = DB.charCodeAt(i) ^ dbMask.charCodeAt(i);
194     }
195 
196     var mask = (0xff00 >> (8 * emLen - emBits)) & 0xff;
197     maskedDB[0] &= ~mask;
198 
199     for (i = 0; i < hLen; i++) {
200         maskedDB.push(H.charCodeAt(i));
201     }
202 
203     maskedDB.push(0xbc);
204 
205     return _zeroPaddingOfSignature(
206             this.doPrivate(new BigInteger(maskedDB)).toString(16),
207             this.n.bitLength());
208 }

214 function _rsasign_getDecryptSignatureBI(biSig, hN, hE) {
215     var rsa = new RSAKey();
216     rsa.setPublic(hN, hE);
217     var biDecryptedSig = rsa.doPublic(biSig);
218     return biDecryptedSig;
219 }

221 function _rsasign_getHexDigestInfoFromSig(biSig, hN, hE) {
222     var biDecryptedSig = _rsasign_getDecryptSignatureBI(biSig, hN, hE);
223     var hDigestInfo = biDecryptedSig.toString(16).replace(/^1f+00/, '');
224     return hDigestInfo;
225 }

227 function _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo) {
228     for (var algName in _RSASIGN_DIHEAD) {
229 	var head = _RSASIGN_DIHEAD[algName];
230 	var len = head.length;
231 	if (hDigestInfo.substring(0, len) == head) {
232 	    var a = [algName, hDigestInfo.substring(len)];
233 	    return a;
234 	}
235     }
236     return [];
237 }

239 function _rsasign_verifySignatureWithArgs(sMsg, biSig, hN, hE) {
240     var hDigestInfo = _rsasign_getHexDigestInfoFromSig(biSig, hN, hE);
241     var digestInfoAry = _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo);
242     if (digestInfoAry.length == 0) return false;
243     var algName = digestInfoAry[0];
244     var diHashValue = digestInfoAry[1];
245     var ff = _RSASIGN_HASHHEXFUNC[algName];
246     var msgHashValue = ff(sMsg);
247     return (diHashValue == msgHashValue);
248 }

250 function _rsasign_verifyHexSignatureForMessage(hSig, sMsg) {
251     var biSig = parseBigInt(hSig, 16);
252     var result = _rsasign_verifySignatureWithArgs(sMsg, biSig,
253 						  this.n.toString(16),
254 						  this.e.toString(16));
255     return result;
256 }

258 /**
259  * verifies a sigature for a message string with RSA public key.<br/>
260  * @name verifyString
261  * @memberOf RSAKey#
262  * @function
263  * @param {String} sMsg message string to be verified.
264  * @param {String} hSig hexadecimal string of siganture.<br/>
265  *                 non-hexadecimal charactors including new lines will be ignored.
266  * @return returns 1 if valid, otherwise 0
267  */
268 function _rsasign_verifyString(sMsg, hSig) {
269     hSig = hSig.replace(_RE_HEXDECONLY, '');
270     if (hSig.length != this.n.bitLength() / 4) return 0;
271     hSig = hSig.replace(/[ \n]+/g, "");
272     var biSig = parseBigInt(hSig, 16);
273     var biDecryptedSig = this.doPublic(biSig);
274     var hDigestInfo = biDecryptedSig.toString(16).replace(/^1f+00/, '');
275     var digestInfoAry = _rsasign_getAlgNameAndHashFromHexDisgestInfo(hDigestInfo);
276   
277     if (digestInfoAry.length == 0) return false;
278     var algName = digestInfoAry[0];
279     var diHashValue = digestInfoAry[1];
280     var ff = _RSASIGN_HASHHEXFUNC[algName];
281     var msgHashValue = ff(sMsg);
282     return (diHashValue == msgHashValue);
283 }

285 /**
286  * verifies a sigature for a message string with RSA public key by PKCS#1 PSS sign.<br/>
287  * @name verifyStringPSS
288  * @memberOf RSAKey#
289  * @function
290  * @param {String} sMsg message string to be verified.
291  * @param {String} hSig hexadecimal string of siganture.<br/>
292  *                 non-hexadecimal charactors including new lines will be ignored.
293  * @return returns 1 if valid, otherwise 0
294  */
295 function _rsasign_verifyStringPSS(sMsg, hSig, hashAlg, sLen) {
296     if (hSig.length !== this.n.bitLength() / 4) {
297         return false;
298     }
299 
300     var hashFunc = _RSASIGN_HASHRAWFUNC[hashAlg];
301     var mHash = hashFunc(sMsg);
302     var hLen = mHash.length;
303     var emBits = this.n.bitLength() - 1;
304     var emLen = Math.ceil(emBits / 8);
305     var i;
306 
307     if (sLen === -1) {
308         sLen = hLen; // same has hash length
309     } else if ((sLen === -2) || (sLen === undefined)) {
310         sLen = emLen - hLen - 2; // maximum
311     } else if (sLen < -2) {
312         throw "invalid salt length";
313     }
314 
315     if (emLen < (hLen + sLen + 2)) {
316         throw "data too long";
317     }
318 
319     var em = this.doPublic(parseBigInt(hSig, 16)).toByteArray();
320 
321     for (i = 0; i < em.length; i += 1) {
322         em[i] &= 0xff;
323     }
324 
325     while (em.length < emLen) {
326         em.unshift(0);
327     }
328 
329     if (em[emLen -1] !== 0xbc) {
330         throw "encoded message does not end in 0xbc";
331     }
332 
333     em = String.fromCharCode.apply(String, em);
334 
335     var maskedDB = em.substr(0, emLen - hLen - 1);
336     var H = em.substr(maskedDB.length, hLen);
337 
338     var mask = (0xff00 >> (8 * emLen - emBits)) & 0xff;
339 
340     if ((maskedDB.charCodeAt(0) & mask) !== 0) {
341         throw "bits beyond keysize not zero";
342     }
343 
344     var dbMask = pss_mgf1_str(H, maskedDB.length, hashFunc);
345     var DB = [];
346 
347     for (i = 0; i < maskedDB.length; i += 1) {
348         DB[i] = maskedDB.charCodeAt(i) ^ dbMask.charCodeAt(i);
349     }
350 
351     DB[0] &= ~mask;
352 
353     var checkLen = emLen - hLen - sLen - 2;
354 
355     for (i = 0; i < checkLen; i += 1) {
356         if (DB[i] !== 0x00) {
357             throw "leftmost octets not zero";
358         }
359     }
360 
361     if (DB[checkLen] !== 0x01) {
362         throw "0x01 marker not found";
363     }
364 
365     return H === hashFunc('\x00\x00\x00\x00\x00\x00\x00\x00' + mHash +
366                           String.fromCharCode.apply(String, DB.slice(-sLen)));
367 }
