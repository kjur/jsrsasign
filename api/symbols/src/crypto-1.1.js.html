 65     /**
 66      * get hexadecimal DigestInfo
 67      * @name getDigestInfoHex
 68      * @memberOf KJUR.crypto.Util
 69      * @function
 70      * @param {String} hHash hexadecimal hash value
 71      * @param {String} alg hash algorithm name (ex. 'sha1')
 72      * @return {String} hexadecimal string DigestInfo ASN.1 structure
 73      */
 74     this.getDigestInfoHex = function(hHash, alg) {
 75 	if (typeof this.DIGESTINFOHEAD[alg] == "undefined")
 76 	    throw "alg not supported in Util.DIGESTINFOHEAD: " + alg;
 77 	return this.DIGESTINFOHEAD[alg] + hHash;
 78     };

 80     /**
 81      * get PKCS#1 padded hexadecimal DigestInfo
 82      * @name getPaddedDigestInfoHex
 83      * @memberOf KJUR.crypto.Util
 84      * @function
 85      * @param {String} hHash hexadecimal hash value
 86      * @param {String} alg hash algorithm name (ex. 'sha1')
 87      * @param {Integer} keySize key bit length (ex. 1024)
 88      * @return {String} hexadecimal string of PKCS#1 padded DigestInfo
 89      */
 90     this.getPaddedDigestInfoHex = function(hHash, alg, keySize) {
 91 	var hDigestInfo = this.getDigestInfoHex(hHash, alg);
 92 	var pmStrLen = keySize / 4; // minimum PM length
 93 
 94 	if (hDigestInfo.length + 22 > pmStrLen) // len(0001+ff(*8)+00+hDigestInfo)=22
 95 	    throw "key is too short for SigAlg: keylen=" + keySize + "," + alg;
 96 
 97 	var hHead = "0001";
 98 	var hTail = "00" + hDigestInfo;
 99 	var hMid = "";
100 	var fLen = pmStrLen - hHead.length - hTail.length;
101 	for (var i = 0; i < fLen; i += 2) {
102 	    hMid += "ff";
103 	}
104 	var hPaddedMessage = hHead + hMid + hTail;
105 	return hPaddedMessage;
106     };

179 /**
180  * MessageDigest class which is very similar to java.security.MessageDigest class
181  * @name KJUR.crypto.MessageDigest
182  * @class MessageDigest class which is very similar to java.security.MessageDigest class
183  * @param {Array} params parameters for constructor
184  * @description
185  * <br/>
186  * Currently this supports following algorithm and providers combination:
187  * <ul>
188  * <li>md5 - cryptojs</li>
189  * <li>sha1 - cryptojs</li>
190  * <li>sha224 - cryptojs</li>
191  * <li>sha256 - cryptojs</li>
192  * <li>sha384 - cryptojs</li>
193  * <li>sha512 - cryptojs</li>
194  * <li>ripemd160 - cryptojs</li>
195  * <li>sha256 - sjcl (NEW from crypto.js 1.0.4)</li>
196  * </ul>
197  * @example
198  * // CryptoJS provider sample
199  * <script src="http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/core.js"></script>
200  * <script src="http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/sha1.js"></script>
201  * <script src="crypto-1.0.js"></script>
202  * var md = new KJUR.crypto.MessageDigest({alg: "sha1", prov: "cryptojs"});
203  * md.updateString('aaa')
204  * var mdHex = md.digest()
205  *
206  * // SJCL(Stanford JavaScript Crypto Library) provider sample
207  * <script src="http://bitwiseshiftleft.github.io/sjcl/sjcl.js"></script>
208  * <script src="crypto-1.0.js"></script>
209  * var md = new KJUR.crypto.MessageDigest({alg: "sha256", prov: "sjcl"}); // sjcl supports sha256 only
210  * md.updateString('aaa')
211  * var mdHex = md.digest()
212  */
213 KJUR.crypto.MessageDigest = function(params) {
214     var md = null;
215     var algName = null;
216     var provName = null;
217     var _CryptoJSMdName = {
218 	'md5': 'CryptoJS.algo.MD5',
219 	'sha1': 'CryptoJS.algo.SHA1',
220 	'sha224': 'CryptoJS.algo.SHA224',
221 	'sha256': 'CryptoJS.algo.SHA256',
222 	'sha384': 'CryptoJS.algo.SHA384',
223 	'sha512': 'CryptoJS.algo.SHA512',
224 	'ripemd160': 'CryptoJS.algo.RIPEMD160'
225     };
226 
227     /**
228      * set hash algorithm and provider
229      * @name setAlgAndProvider
230      * @memberOf KJUR.crypto.MessageDigest
231      * @function
232      * @param {String} alg hash algorithm name
233      * @param {String} prov provider name
234      * @description
235      * @example
236      * // for SHA1
237      * md.setAlgAndProvider('sha1', 'cryptojs');
238      * // for RIPEMD160
239      * md.setAlgAndProvider('ripemd160', 'cryptojs');
240      */
241     this.setAlgAndProvider = function(alg, prov) {
242 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(alg) != -1 &&
243 	    prov == 'cryptojs') {
244 	    try {
245 		this.md = eval(_CryptoJSMdName[alg]).create();
246 	    } catch (ex) {
247 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
248 	    }
249 	    this.updateString = function(str) {
250 		this.md.update(str);
251 	    };
252 	    this.updateHex = function(hex) {
253 		var wHex = CryptoJS.enc.Hex.parse(hex);
254 		this.md.update(wHex);
255 	    };
256 	    this.digest = function() {
257 		var hash = this.md.finalize();
258 		return hash.toString(CryptoJS.enc.Hex);
259 	    };
260 	    this.digestString = function(str) {
261 		this.updateString(str);
262 		return this.digest();
263 	    };
264 	    this.digestHex = function(hex) {
265 		this.updateHex(hex);
266 		return this.digest();
267 	    };
268 	}
269 	if (':sha256:'.indexOf(alg) != -1 &&
270 	    prov == 'sjcl') {
271 	    try {
272 		this.md = new sjcl.hash.sha256();
273 	    } catch (ex) {
274 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
275 	    }
276 	    this.updateString = function(str) {
277 		this.md.update(str);
278 	    };
279 	    this.updateHex = function(hex) {
280 		var baHex = sjcl.codec.hex.toBits(hex);
281 		this.md.update(baHex);
282 	    };
283 	    this.digest = function() {
284 		var hash = this.md.finalize();
285 		return sjcl.codec.hex.fromBits(hash);
286 	    };
287 	    this.digestString = function(str) {
288 		this.updateString(str);
289 		return this.digest();
290 	    };
291 	    this.digestHex = function(hex) {
292 		this.updateHex(hex);
293 		return this.digest();
294 	    };
295 	}
296     };
297

299      * update digest by specified string
300      * @name updateString

302      * @function
303      * @param {String} str string to update

306      * md.updateString('New York');
307      */

310     };

313      * update digest by specified hexadecimal string
314      * @name updateHex
315      * @memberOf KJUR.crypto.MessageDigest

318      * @description
319      * @example
320      * md.updateHex('0afe36');
321      */
322     this.updateHex = function(hex) {
323 	throw "updateHex(hex) not supported for this alg/prov: " + this.algName + "/" + this.provName;
324     };
325 
326     /**
327      * completes hash calculation and returns hash result
328      * @name digest
329      * @memberOf KJUR.crypto.MessageDigest
330      * @function
331      * @description
332      * @example
333      * md.digest()
334      */
335     this.digest = function() {
336 	throw "digest() not supported for this alg/prov: " + this.algName + "/" + this.provName;
337     };
338 
339     /**
340      * performs final update on the digest using string, then completes the digest computation
341      * @name digestString
342      * @memberOf KJUR.crypto.MessageDigest
343      * @function
344      * @param {String} str string to final update
345      * @description
346      * @example
347      * md.digestString('aaa')
348      */
349     this.digestString = function(str) {
350 	throw "digestString(str) not supported for this alg/prov: " + this.algName + "/" + this.provName;
351     };
352 
353     /**
354      * performs final update on the digest using hexadecimal string, then completes the digest computation
355      * @name digestHex
356      * @memberOf KJUR.crypto.MessageDigest
357      * @function
358      * @param {String} hex hexadecimal string to final update
359      * @description
360      * @example
361      * md.digestHex('0f2abd')
362      */
363     this.digestHex = function(hex) {
364 	throw "digestHex(hex) not supported for this alg/prov: " + this.algName + "/" + this.provName;
365     };
366 
367     if (typeof params != "undefined") {
368 	if (typeof params['alg'] != "undefined") {
369 	    this.algName = params['alg'];
370 	    this.provName = params['prov'];
371 	    this.setAlgAndProvider(params['alg'], params['prov']);
372 	}
373     }
374 };

377 /**
378  * Signature class which is very similar to java.security.Signature class
379  * @name KJUR.crypto.Signature
380  * @class Signature class which is very similar to java.security.Signature class
381  * @param {Array} params parameters for constructor
382  * @property {String} state Current state of this signature object whether 'SIGN', 'VERIFY' or null
383  * @description
384  * <br/>
385  * As for params of constructor's argument, it can be specify following attributes:
386  * <ul>
387  * <li>alg - signature algorithm name (ex. {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}withRSA)</li>
388  * <li>provider - currently 'cryptojs/jsrsa' only</li>
389  * <li>prvkeypem - PEM string of signer's private key. If this specified, no need to call initSign(prvKey).</li>
390  * </ul>
391  * <h4>SUPPORTED ALGORITHMS AND PROVIDERS</h4>
392  * This Signature class supports following signature algorithm and provider names:
393  * <ul>
394  * <li>MD5withRSA - cryptojs/jsrsa</li>
395  * <li>SHA1withRSA - cryptojs/jsrsa</li>
396  * <li>SHA224withRSA - cryptojs/jsrsa</li>
397  * <li>SHA256withRSA - cryptojs/jsrsa</li>
398  * <li>SHA384withRSA - cryptojs/jsrsa</li>
399  * <li>SHA512withRSA - cryptojs/jsrsa</li>
400  * <li>RIPEMD160withRSA - cryptojs/jsrsa</li>
401  * <li>MD5withECDSA - cryptojs/jsrsa</li>
402  * <li>SHA1withECDSA - cryptojs/jsrsa</li>
403  * <li>SHA224withECDSA - cryptojs/jsrsa</li>
404  * <li>SHA256withECDSA - cryptojs/jsrsa</li>
405  * <li>SHA384withECDSA - cryptojs/jsrsa</li>
406  * <li>SHA512withECDSA - cryptojs/jsrsa</li>
407  * <li>RIPEMD160withECDSA - cryptojs/jsrsa</li>
408  * </ul>
409  * Here are supported elliptic cryptographic curve names and their aliases for ECDSA:
410  * <ul>
411  * <li>secp256k1</li>
412  * <li>secp256r1, NIST P-256, P-256, prime256v1</li>
413  * <li>secp384r1, NIST P-384, P-384</li>
414  * </ul>
415  * <h4>EXAMPLES</h4>
416  * @example
417  * // RSA signature generation
418  * var sig = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
419  * sig.initSign(prvKey);
420  * sig.updateString('aaa');
421  * var hSigVal = sig.sign();
422  *
423  * // RSA signature validation
424  * var sig2 = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
425  * sig2.initVerifyByCertificatePEM(cert)
426  * sig.updateString('aaa');
427  * var isValid = sig2.verify(hSigVal);
428  * 
429  * // EC key generation
430  * var ec = new KJUR.crypto.ECDSA({'curve': 'secp256r1'});
431  * var keypair = ec.generateKeyPairHex();
432  *
433  * // ECDSA signing
434  * var sig = new KJUR.crypto.Signature({'alg':'SHA1withECDSA', 'prov':'cryptojs/jsrsa'});
435  * sig.initSign({'ecprvhex': keypair.ecprvhex, 'eccurvename': 'secp256r1'});
436  * sig.updateString('aaa');
437  * var sigValueHex = sig.sign();
438  *
439  * // ECDSA verifying
440  * var sig2 = new KJUR.crypto.Signature({'alg':'SHA1withECDSA', 'prov':'cryptojs/jsrsa'});
441  * sig.initVerifyByPublicKey({'ecpubhex': keypair.ecpubhex, 'eccurvename': 'secp256r1'});
442  * sig.updateString('aaa');
443  * var isValid = sig.verify(sigValueHex);
444  */
445 KJUR.crypto.Signature = function(params) {
446     var prvKey = null; // RSAKey for signing
447     var pubKey = null; // RSAKey for verifying
448 
449     var md = null; // KJUR.crypto.MessageDigest object
450     var sig = null;
451     var algName = null;
452     var provName = null;
453     var algProvName = null;
454     var mdAlgName = null;
455     var pubkeyAlgName = null;
456     var state = null;
457 
458     var sHashHex = null; // hex hash value for hex
459     var hDigestInfo = null;
460     var hPaddedDigestInfo = null;
461     var hSign = null;
462 
463     this._setAlgNames = function() {
464 	if (this.algName.match(/^(.+)with(.+)$/)) {
465 	    this.mdAlgName = RegExp.$1.toLowerCase();
466 	    this.pubkeyAlgName = RegExp.$2.toLowerCase();
467 	}
468     };
469 
470     this._zeroPaddingOfSignature = function(hex, bitLength) {
471 	var s = "";
472 	var nZero = bitLength / 4 - hex.length;
473 	for (var i = 0; i < nZero; i++) {
474 	    s = s + "0";
475 	}
476 	return s + hex;
477     };
478 
479     /**
480      * set signature algorithm and provider
481      * @name setAlgAndProvider
482      * @memberOf KJUR.crypto.Signature
483      * @function
484      * @param {String} alg signature algorithm name
485      * @param {String} prov provider name
486      * @description
487      * @example
488      * md.setAlgAndProvider('SHA1withRSA', 'cryptojs/jsrsa');
489      */
490     this.setAlgAndProvider = function(alg, prov) {
491 	this._setAlgNames();
492 	if (prov != 'cryptojs/jsrsa')
493 	    throw "provider not supported: " + prov;
494 
495 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(this.mdAlgName) != -1) {
496 	    try {
497 		this.md = new KJUR.crypto.MessageDigest({'alg':this.mdAlgName,'prov':'cryptojs'});
498 	    } catch (ex) {
499 		throw "setAlgAndProvider hash alg set fail alg=" + this.mdAlgName + "/" + ex;
500 	    }
501 
502 	    this.initSign = function(params) {
503 		if (typeof params['ecprvhex'] == 'string' && typeof params['eccurvename'] == 'string') {
504 		    this.ecprvhex = params['ecprvhex'];
505 		    this.eccurvename = params['eccurvename'];
506 		} else {
507 		    this.prvKey = params;
508 		}
509 		this.state = "SIGN";
510 	    };
511 
512 	    this.initVerifyByPublicKey = function(params) {
513 		if (typeof params['ecpubhex'] == 'string' &&
514 		    typeof params['eccurvename'] == 'string') {
515 		    this.ecpubhex = params['ecpubhex'];
516 		    this.eccurvename = params['eccurvename'];
517 		} else {
518 		    this.pubKey = params;
519 		}
520 		this.state = "VERIFY";
521 	    };
522 
523 	    this.initVerifyByCertificatePEM = function(certPEM) {
524 		var x509 = new X509();
525 		x509.readCertPEM(certPEM);
526 		this.pubKey = x509.subjectPublicKeyRSA;
527 		this.state = "VERIFY";
528 	    };
529 
530 	    this.updateString = function(str) {
531 		this.md.updateString(str);
532 	    };
533 	    this.updateHex = function(hex) {
534 		this.md.updateHex(hex);
535 	    };
536 
537 	    this.sign = function() {
538 		if (typeof this.ecprvhex != "undefined" &&
539 		    typeof this.eccurvename != "undefined") {
540 		    var ec = new KJUR.crypto.ECDSA({'curve': this.eccurvename});
541 		    this.sHashHex = this.md.digest();
542 		    this.hSign = ec.signHex(this.sHashHex, this.ecprvhex);
543 		    return this.hSign;
544 		} else {
545 		    var util = KJUR.crypto.Util;
546 		    var keyLen = this.prvKey.n.bitLength();
547 		    this.sHashHex = this.md.digest();
548 		    this.hDigestInfo = util.getDigestInfoHex(this.sHashHex, this.mdAlgName);
549 		    this.hPaddedDigestInfo = 
550                         util.getPaddedDigestInfoHex(this.sHashHex, this.mdAlgName, keyLen);
551 
552 		    var biPaddedDigestInfo = parseBigInt(this.hPaddedDigestInfo, 16);
553 		    this.hoge = biPaddedDigestInfo.toString(16);
554 
555 		    var biSign = this.prvKey.doPrivate(biPaddedDigestInfo);
556 		    this.hSign = this._zeroPaddingOfSignature(biSign.toString(16), keyLen);
557 		    return this.hSign;
558 		}
559 	    };
560 	    this.signString = function(str) {
561 		this.updateString(str);
562 		this.sign();
563 	    };
564 	    this.signHex = function(hex) {
565 		this.updateHex(hex);
566 		this.sign();
567 	    };
568 	    this.verify = function(hSigVal) {
569 		if (typeof this.ecpubhex != "undefined" &&
570 		    typeof this.eccurvename != "undefined") {
571 		    this.sHashHex = this.md.digest();
572 		    var ec = new KJUR.crypto.ECDSA({curve: this.eccurvename});
573 		    return ec.verifyHex(this.sHashHex, hSigVal, this.ecpubhex);
574 		} else {
575 		    var util = KJUR.crypto.Util;
576 		    var keyLen = this.pubKey.n.bitLength();
577 		    this.sHashHex = this.md.digest();
578 
579 		    var biSigVal = parseBigInt(hSigVal, 16);
580 		    var biPaddedDigestInfo = this.pubKey.doPublic(biSigVal);
581 		    this.hPaddedDigestInfo = biPaddedDigestInfo.toString(16);
582 		    var s = this.hPaddedDigestInfo;
583 		    s = s.replace(/^1ff+00/, '');
584 
585 		    var hDIHEAD = KJUR.crypto.Util.DIGESTINFOHEAD[this.mdAlgName];
586 		    if (s.indexOf(hDIHEAD) != 0) {
587 			return false;
588 		    }
589 		    var hHashFromDI = s.substr(hDIHEAD.length);
590 		    //alert(hHashFromDI + "\n" + this.sHashHex);
591 		    return (hHashFromDI == this.sHashHex);
592 		}
593 	    };
594 	}
