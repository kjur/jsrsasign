 94     /**
 95      * get hexadecimal DigestInfo
 96      * @name getDigestInfoHex
 97      * @memberOf KJUR.crypto.Util
 98      * @function
 99      * @param {String} hHash hexadecimal hash value
100      * @param {String} alg hash algorithm name (ex. 'sha1')
101      * @return {String} hexadecimal string DigestInfo ASN.1 structure
102      */
103     this.getDigestInfoHex = function(hHash, alg) {
104 	if (typeof this.DIGESTINFOHEAD[alg] == "undefined")
105 	    throw "alg not supported in Util.DIGESTINFOHEAD: " + alg;
106 	return this.DIGESTINFOHEAD[alg] + hHash;
107     };

109     /**
110      * get PKCS#1 padded hexadecimal DigestInfo
111      * @name getPaddedDigestInfoHex
112      * @memberOf KJUR.crypto.Util
113      * @function
114      * @param {String} hHash hexadecimal hash value
115      * @param {String} alg hash algorithm name (ex. 'sha1')
116      * @param {Integer} keySize key bit length (ex. 1024)
117      * @return {String} hexadecimal string of PKCS#1 padded DigestInfo
118      */
119     this.getPaddedDigestInfoHex = function(hHash, alg, keySize) {
120 	var hDigestInfo = this.getDigestInfoHex(hHash, alg);
121 	var pmStrLen = keySize / 4; // minimum PM length
122 
123 	if (hDigestInfo.length + 22 > pmStrLen) // len(0001+ff(*8)+00+hDigestInfo)=22
124 	    throw "key is too short for SigAlg: keylen=" + keySize + "," + alg;
125 
126 	var hHead = "0001";
127 	var hTail = "00" + hDigestInfo;
128 	var hMid = "";
129 	var fLen = pmStrLen - hHead.length - hTail.length;
130 	for (var i = 0; i < fLen; i += 2) {
131 	    hMid += "ff";
132 	}
133 	var hPaddedMessage = hHead + hMid + hTail;
134 	return hPaddedMessage;
135     };

248 /**
249  * MessageDigest class which is very similar to java.security.MessageDigest class
250  * @name KJUR.crypto.MessageDigest
251  * @class MessageDigest class which is very similar to java.security.MessageDigest class
252  * @param {Array} params parameters for constructor
253  * @description
254  * <br/>
255  * Currently this supports following algorithm and providers combination:
256  * <ul>
257  * <li>md5 - cryptojs</li>
258  * <li>sha1 - cryptojs</li>
259  * <li>sha224 - cryptojs</li>
260  * <li>sha256 - cryptojs</li>
261  * <li>sha384 - cryptojs</li>
262  * <li>sha512 - cryptojs</li>
263  * <li>ripemd160 - cryptojs</li>
264  * <li>sha256 - sjcl (NEW from crypto.js 1.0.4)</li>
265  * </ul>
266  * @example
267  * // CryptoJS provider sample
268  * <script src="http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/core.js"></script>
269  * <script src="http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/sha1.js"></script>
270  * <script src="crypto-1.0.js"></script>
271  * var md = new KJUR.crypto.MessageDigest({alg: "sha1", prov: "cryptojs"});
272  * md.updateString('aaa')
273  * var mdHex = md.digest()
274  *
275  * // SJCL(Stanford JavaScript Crypto Library) provider sample
276  * <script src="http://bitwiseshiftleft.github.io/sjcl/sjcl.js"></script>
277  * <script src="crypto-1.0.js"></script>
278  * var md = new KJUR.crypto.MessageDigest({alg: "sha256", prov: "sjcl"}); // sjcl supports sha256 only
279  * md.updateString('aaa')
280  * var mdHex = md.digest()
281  */
282 KJUR.crypto.MessageDigest = function(params) {
283     var md = null;
284     var algName = null;
285     var provName = null;
286     var _CryptoJSMdName = {
287 	'md5': 'CryptoJS.algo.MD5',
288 	'sha1': 'CryptoJS.algo.SHA1',
289 	'sha224': 'CryptoJS.algo.SHA224',
290 	'sha256': 'CryptoJS.algo.SHA256',
291 	'sha384': 'CryptoJS.algo.SHA384',
292 	'sha512': 'CryptoJS.algo.SHA512',
293 	'ripemd160': 'CryptoJS.algo.RIPEMD160'
294     };
295 
296     /**
297      * set hash algorithm and provider
298      * @name setAlgAndProvider
299      * @memberOf KJUR.crypto.MessageDigest
300      * @function
301      * @param {String} alg hash algorithm name
302      * @param {String} prov provider name
303      * @description
304      * @example
305      * // for SHA1
306      * md.setAlgAndProvider('sha1', 'cryptojs');
307      * // for RIPEMD160
308      * md.setAlgAndProvider('ripemd160', 'cryptojs');
309      */
310     this.setAlgAndProvider = function(alg, prov) {
311 	if (alg != null && prov === undefined) prov = KJUR.crypto.Util.DEFAULTPROVIDER[alg];
312 
313 	// for cryptojs
314 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(alg) != -1 &&
315 	    prov == 'cryptojs') {
316 	    try {
317 		this.md = eval(_CryptoJSMdName[alg]).create();
318 	    } catch (ex) {
319 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
320 	    }
321 	    this.updateString = function(str) {
322 		this.md.update(str);
323 	    };
324 	    this.updateHex = function(hex) {
325 		var wHex = CryptoJS.enc.Hex.parse(hex);
326 		this.md.update(wHex);
327 	    };
328 	    this.digest = function() {
329 		var hash = this.md.finalize();
330 		return hash.toString(CryptoJS.enc.Hex);
331 	    };
332 	    this.digestString = function(str) {
333 		this.updateString(str);
334 		return this.digest();
335 	    };
336 	    this.digestHex = function(hex) {
337 		this.updateHex(hex);
338 		return this.digest();
339 	    };
340 	}
341 	if (':sha256:'.indexOf(alg) != -1 &&
342 	    prov == 'sjcl') {
343 	    try {
344 		this.md = new sjcl.hash.sha256();
345 	    } catch (ex) {
346 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
347 	    }
348 	    this.updateString = function(str) {
349 		this.md.update(str);
350 	    };
351 	    this.updateHex = function(hex) {
352 		var baHex = sjcl.codec.hex.toBits(hex);
353 		this.md.update(baHex);
354 	    };
355 	    this.digest = function() {
356 		var hash = this.md.finalize();
357 		return sjcl.codec.hex.fromBits(hash);
358 	    };
359 	    this.digestString = function(str) {
360 		this.updateString(str);
361 		return this.digest();
362 	    };
363 	    this.digestHex = function(hex) {
364 		this.updateHex(hex);
365 		return this.digest();
366 	    };
367 	}
368     };
369

371      * update digest by specified string

390      * @description

396     };
397 
398     /**
399      * completes hash calculation and returns hash result
400      * @name digest
401      * @memberOf KJUR.crypto.MessageDigest
402      * @function
403      * @description
404      * @example
405      * md.digest()
406      */
407     this.digest = function() {
408 	throw "digest() not supported for this alg/prov: " + this.algName + "/" + this.provName;
409     };
410 
411     /**
412      * performs final update on the digest using string, then completes the digest computation
413      * @name digestString
414      * @memberOf KJUR.crypto.MessageDigest
415      * @function
416      * @param {String} str string to final update
417      * @description
418      * @example
419      * md.digestString('aaa')
420      */
421     this.digestString = function(str) {
422 	throw "digestString(str) not supported for this alg/prov: " + this.algName + "/" + this.provName;
423     };
424 
425     /**
426      * performs final update on the digest using hexadecimal string, then completes the digest computation
427      * @name digestHex
428      * @memberOf KJUR.crypto.MessageDigest
429      * @function
430      * @param {String} hex hexadecimal string to final update
431      * @description
432      * @example
433      * md.digestHex('0f2abd')
434      */
435     this.digestHex = function(hex) {
436 	throw "digestHex(hex) not supported for this alg/prov: " + this.algName + "/" + this.provName;
437     };
438 
439     if (params !== undefined) {
440 	if (params['alg'] !== undefined) {
441 	    this.algName = params['alg'];
442 	    if (params['prov'] === undefined)
443 		this.provName = KJUR.crypto.Util.DEFAULTPROVIDER[this.algName];
444 	    this.setAlgAndProvider(this.algName, this.provName);
445 	}
446     }
447 };

450 /**
451  * Signature class which is very similar to java.security.Signature class
452  * @name KJUR.crypto.Signature
453  * @class Signature class which is very similar to java.security.Signature class
454  * @param {Array} params parameters for constructor
455  * @property {String} state Current state of this signature object whether 'SIGN', 'VERIFY' or null
456  * @description
457  * <br/>
458  * As for params of constructor's argument, it can be specify following attributes:
459  * <ul>
460  * <li>alg - signature algorithm name (ex. {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}withRSA)</li>
461  * <li>provider - currently 'cryptojs/jsrsa' only</li>
462  * <li>prvkeypem - PEM string of signer's private key. If this specified, no need to call initSign(prvKey).</li>
463  * </ul>
464  * <h4>SUPPORTED ALGORITHMS AND PROVIDERS</h4>
465  * This Signature class supports following signature algorithm and provider names:
466  * <ul>
467  * <li>MD5withRSA - cryptojs/jsrsa</li>
468  * <li>SHA1withRSA - cryptojs/jsrsa</li>
469  * <li>SHA224withRSA - cryptojs/jsrsa</li>
470  * <li>SHA256withRSA - cryptojs/jsrsa</li>
471  * <li>SHA384withRSA - cryptojs/jsrsa</li>
472  * <li>SHA512withRSA - cryptojs/jsrsa</li>
473  * <li>RIPEMD160withRSA - cryptojs/jsrsa</li>
474  * <li>MD5withECDSA - cryptojs/jsrsa</li>
475  * <li>SHA1withECDSA - cryptojs/jsrsa</li>
476  * <li>SHA224withECDSA - cryptojs/jsrsa</li>
477  * <li>SHA256withECDSA - cryptojs/jsrsa</li>
478  * <li>SHA384withECDSA - cryptojs/jsrsa</li>
479  * <li>SHA512withECDSA - cryptojs/jsrsa</li>
480  * <li>RIPEMD160withECDSA - cryptojs/jsrsa</li>
481  * </ul>
482  * Here are supported elliptic cryptographic curve names and their aliases for ECDSA:
483  * <ul>
484  * <li>secp256k1</li>
485  * <li>secp256r1, NIST P-256, P-256, prime256v1</li>
486  * <li>secp384r1, NIST P-384, P-384</li>
487  * </ul>
488  * <h4>EXAMPLES</h4>
489  * @example
490  * // RSA signature generation
491  * var sig = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
492  * sig.initSign(prvKey);
493  * sig.updateString('aaa');
494  * var hSigVal = sig.sign();
495  *
496  * // RSA signature validation
497  * var sig2 = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
498  * sig2.initVerifyByCertificatePEM(cert)
499  * sig.updateString('aaa');
500  * var isValid = sig2.verify(hSigVal);
501  * 
502  * // EC key generation
503  * var ec = new KJUR.crypto.ECDSA({'curve': 'secp256r1'});
504  * var keypair = ec.generateKeyPairHex();
505  *
506  * // ECDSA signing
507  * var sig = new KJUR.crypto.Signature({'alg':'SHA1withECDSA', 'prov':'cryptojs/jsrsa'});
508  * sig.initSign({'ecprvhex': keypair.ecprvhex, 'eccurvename': 'secp256r1'});
509  * sig.updateString('aaa');
510  * var sigValueHex = sig.sign();
511  *
512  * // ECDSA verifying
513  * var sig2 = new KJUR.crypto.Signature({'alg':'SHA1withECDSA', 'prov':'cryptojs/jsrsa'});
514  * sig.initVerifyByPublicKey({'ecpubhex': keypair.ecpubhex, 'eccurvename': 'secp256r1'});
515  * sig.updateString('aaa');
516  * var isValid = sig.verify(sigValueHex);
517  */
518 KJUR.crypto.Signature = function(params) {
519     var prvKey = null; // RSAKey for signing
520     var pubKey = null; // RSAKey for verifying
521 
522     var md = null; // KJUR.crypto.MessageDigest object
523     var sig = null;
524     var algName = null;
525     var provName = null;
526     var algProvName = null;
527     var mdAlgName = null;
528     var pubkeyAlgName = null;
529     var state = null;
530 
531     var sHashHex = null; // hex hash value for hex
532     var hDigestInfo = null;
533     var hPaddedDigestInfo = null;
534     var hSign = null;
535 
536     this._setAlgNames = function() {
537 	if (this.algName.match(/^(.+)with(.+)$/)) {
538 	    this.mdAlgName = RegExp.$1.toLowerCase();
539 	    this.pubkeyAlgName = RegExp.$2.toLowerCase();
540 	}
541     };
542 
543     this._zeroPaddingOfSignature = function(hex, bitLength) {
544 	var s = "";
545 	var nZero = bitLength / 4 - hex.length;
546 	for (var i = 0; i < nZero; i++) {
547 	    s = s + "0";
548 	}
549 	return s + hex;
550     };
551 
552     /**
553      * set signature algorithm and provider
554      * @name setAlgAndProvider
555      * @memberOf KJUR.crypto.Signature
556      * @function
557      * @param {String} alg signature algorithm name
558      * @param {String} prov provider name
559      * @description
560      * @example
561      * md.setAlgAndProvider('SHA1withRSA', 'cryptojs/jsrsa');
562      */
563     this.setAlgAndProvider = function(alg, prov) {
564 	this._setAlgNames();
565 	if (prov != 'cryptojs/jsrsa')
566 	    throw "provider not supported: " + prov;
567 
568 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(this.mdAlgName) != -1) {
569 	    try {
570 		this.md = new KJUR.crypto.MessageDigest({'alg':this.mdAlgName,'prov':'cryptojs'});
571 	    } catch (ex) {
572 		throw "setAlgAndProvider hash alg set fail alg=" + this.mdAlgName + "/" + ex;
573 	    }
574 
575 	    this.initSign = function(params) {
576 		if (typeof params['ecprvhex'] == 'string' && typeof params['eccurvename'] == 'string') {
577 		    this.ecprvhex = params['ecprvhex'];
578 		    this.eccurvename = params['eccurvename'];
579 		} else {
580 		    this.prvKey = params;
581 		}
582 		this.state = "SIGN";
583 	    };
584

610 	    this.sign = function() {
611 		if (typeof this.ecprvhex != "undefined" &&
612 		    typeof this.eccurvename != "undefined") {
613 		    var ec = new KJUR.crypto.ECDSA({'curve': this.eccurvename});
614 		    this.sHashHex = this.md.digest();
615 		    this.hSign = ec.signHex(this.sHashHex, this.ecprvhex);
616 		    return this.hSign;
617 		} else {
618 		    var util = KJUR.crypto.Util;
619 		    var keyLen = this.prvKey.n.bitLength();
620 		    this.sHashHex = this.md.digest();
621 		    this.hDigestInfo = util.getDigestInfoHex(this.sHashHex, this.mdAlgName);
622 		    this.hPaddedDigestInfo = 
623                         util.getPaddedDigestInfoHex(this.sHashHex, this.mdAlgName, keyLen);
624 
625 		    var biPaddedDigestInfo = parseBigInt(this.hPaddedDigestInfo, 16);
626 		    this.hoge = biPaddedDigestInfo.toString(16);
627 
628 		    var biSign = this.prvKey.doPrivate(biPaddedDigestInfo);
629 		    this.hSign = this._zeroPaddingOfSignature(biSign.toString(16), keyLen);
630 		    return this.hSign;
631 		}
632 	    };
633 	    this.signString = function(str) {
634 		this.updateString(str);
635 		this.sign();
636 	    };
637 	    this.signHex = function(hex) {
638 		this.updateHex(hex);
639 		this.sign();
640 	    };
641 	    this.verify = function(hSigVal) {
642 		if (typeof this.ecpubhex != "undefined" &&
643 		    typeof this.eccurvename != "undefined") {
644 		    this.sHashHex = this.md.digest();
645 		    var ec = new KJUR.crypto.ECDSA({curve: this.eccurvename});
646 		    return ec.verifyHex(this.sHashHex, hSigVal, this.ecpubhex);
647 		} else {
648 		    var util = KJUR.crypto.Util;
649 		    var keyLen = this.pubKey.n.bitLength();
650 		    this.sHashHex = this.md.digest();
651 
652 		    var biSigVal = parseBigInt(hSigVal, 16);
653 		    var biPaddedDigestInfo = this.pubKey.doPublic(biSigVal);
654 		    this.hPaddedDigestInfo = biPaddedDigestInfo.toString(16);
655 		    var s = this.hPaddedDigestInfo;
656 		    s = s.replace(/^1ff+00/, '');
657 
658 		    var hDIHEAD = KJUR.crypto.Util.DIGESTINFOHEAD[this.mdAlgName];
659 		    if (s.indexOf(hDIHEAD) != 0) {
660 			return false;
661 		    }
662 		    var hHashFromDI = s.substr(hDIHEAD.length);
663 		    //alert(hHashFromDI + "\n" + this.sHashHex);
664 		    return (hHashFromDI == this.sHashHex);
665 		}
666 	    };
667 	}
