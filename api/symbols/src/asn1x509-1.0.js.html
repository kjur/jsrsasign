 83 /**
 84  * X.509 Certificate class to sign and generate hex encoded certificate
 85  * @name KJUR.asn1.x509.Certificate
 86  * @class X.509 Certificate class to sign and generate hex encoded certificate
 87  * @param {Array} params associative array of parameters (ex. {'tbscertobj': obj, 'rsaprvkey': key})
 88  * @extends KJUR.asn1.ASN1Object
 89  * @description
 90  * <br/>
 91  * As for argument 'params' for constructor, you can specify one of
 92  * following properties:
 93  * <ul>
 94  * <li>tbscertobj - specify {@link KJUR.asn1.x509.TBSCertificate} object</li>
 95  * <li>rsaprvkey - specify {@link RSAKey} object CA private key</li>
 96  * </ul>
 97  * NOTE: 'params' can be omitted.
 98  * <h4>EXAMPLE</h4>
 99  * @example
100  * var prvKey = new RSAKey(); // CA's private key
101  * prvKey.readPrivateKeyFromASN1HexString("3080...");
102  * var cert = new KJUR.asn1x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
103  * cert.sign(); // issue certificate by CA's private key
104  * var hCert = cert.getEncodedHex();
105  *
106  * // Certificate  ::=  SEQUENCE  {
107  * //     tbsCertificate       TBSCertificate,
108  * //     signatureAlgorithm   AlgorithmIdentifier,
109  * //     signature            BIT STRING  }	    
110  */
111 KJUR.asn1.x509.Certificate = function(params) {
112     KJUR.asn1.x509.Certificate.superclass.constructor.call(this);
113     var asn1TBSCert = null;
114     var asn1SignatureAlg = null;
115     var asn1Sig = null;
116     var hexSig = null;
117     var rsaPrvKey = null;
118 
119     
120     /**
121      * set PKCS#5 encrypted RSA PEM private key as CA key
122      * @name setRsaPrvKeyByPEMandPass
123      * @memberOf KJUR.asn1.x509.Certificate
124      * @function
125      * @param {String} rsaPEM string of PKCS#5 encrypted RSA PEM private key
126      * @param {String} passPEM passcode string to decrypt private key
127      * @since 1.0.1
128      * @description
129      * <br/>
130      * <h4>EXAMPLES</h4>
131      * @example
132      * var cert = new KJUR.asn1.x509.Certificate({'tbscertobj': tbs});
133      * cert.setRsaPrvKeyByPEMandPass("-----BEGIN RSA PRIVATE..(snip)", "password");
134      */
135     this.setRsaPrvKeyByPEMandPass = function(rsaPEM, passPEM) {
136 	var caKeyHex = PKCS5PKEY.getDecryptedKeyHex(rsaPEM, passPEM);
137 	var caKey = new RSAKey();
138 	caKey.readPrivateKeyFromASN1HexString(caKeyHex);  
139 	this.rsaPrvKey = caKey;
140     };
141 
142     /**
143      * sign TBSCertificate and set signature value internally
144      * @name sign
145      * @memberOf KJUR.asn1.x509.Certificate
146      * @function
147      * @description
148      * @example
149      * var cert = new KJUR.asn1.x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
150      * cert.sign();
151      */
152     this.sign = function() {
153 	this.asn1SignatureAlg = this.asn1TBSCert.asn1SignatureAlg;
154 
155 	sig = new KJUR.crypto.Signature({'alg': 'SHA1withRSA', 'prov': 'cryptojs/jsrsa'});
156 	sig.initSign(this.rsaPrvKey);
157 	sig.updateHex(this.asn1TBSCert.getEncodedHex());
158 	this.hexSig = sig.sign();
159 
160 	this.asn1Sig = new KJUR.asn1.DERBitString({'hex': '00' + this.hexSig});
161 	
162 	var seq = new KJUR.asn1.DERSequence({'array': [this.asn1TBSCert,
163 						       this.asn1SignatureAlg,
164 						       this.asn1Sig]});
165 	this.hTLV = seq.getEncodedHex();
166 	this.isModified = false;
167     };
168 
169     this.getEncodedHex = function() {
170 	if (this.isModified == false && this.hTLV != null) return this.hTLV;
171 	throw "not signed yet";
172     };
173 
174     /**
175      * get PEM formatted certificate string after signed
176      * @name getPEMString
177      * @memberOf KJUR.asn1.x509.Certificate
178      * @function
179      * @return PEM formatted string of certificate
180      * @description
181      * @example
182      * var cert = new KJUR.asn1.x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
183      * cert.sign();
184      * var sPEM =  cert.getPEMString();
185      */
186     this.getPEMString = function() {
187 	var hCert = this.getEncodedHex();
188 	var wCert = CryptoJS.enc.Hex.parse(hCert);
189 	var b64Cert = CryptoJS.enc.Base64.stringify(wCert);
190 	var pemBody = b64Cert.replace(/(.{64})/g, "$1\r\n");
191 	return "-----BEGIN CERTIFICATE-----\r\n" + pemBody + "\r\n-----END CERTIFICATE-----\r\n";
192     };
193 
194     if (typeof params != "undefined") {
195 	if (typeof params['tbscertobj'] != "undefined") {
196 	    this.asn1TBSCert = params['tbscertobj'];
197 	}
198 	if (typeof params['rsaprvkey'] != "undefined") {
199 	    this.rsaPrvKey = params['rsaprvkey'];
200 	}
201 	if ((typeof params['rsaprvpem'] != "undefined") &&
202 	    (typeof params['rsaprvpas'] != "undefined")) {
203 	    this.setRsaPrvKeyByPEMandPass(params['rsaprvpem'], params['rsaprvpas']);
204 	}
205     }
206 };

211  * @name KJUR.asn1.x509.TBSCertificate

213  * @param {Array} params associative array of parameters (ex. {})
214  * @extends KJUR.asn1.ASN1Object
215  * @description
216  * <br/>
217  * <h4>EXAMPLE</h4>
218  * @example
219  *  var o = new KJUR.asn1.x509.TBSCertificate();
220  *  o.setSerialNumberByParam({'int': 4});

226  *  o.setSubjectPublicKeyByParam({'rsakey': rsaKey});
227  *  o.appendExtension(new KJUR.asn1.x509.BasicConstraints({'cA':true}));
228  *  o.appendExtension(new KJUR.asn1.x509.KeyUsage({'bin':'11'}));

231     KJUR.asn1.x509.TBSCertificate.superclass.constructor.call(this);

234 	this.asn1Array = new Array();
235 
236 	this.asn1Version =

238 	this.asn1SerialNumber = null;
239 	this.asn1SignatureAlg = null;
240 	this.asn1Issuer = null;
241 	this.asn1NotBefore = null;
242 	this.asn1NotAfter = null;
243 	this.asn1Subject = null;
244 	this.asn1SubjPKey = null;
245 	this.extensionsArray = new Array();
246     };

293      * @name setNotBeforeByParam
294      * @memberOf KJUR.asn1.x509.TBSCertificate

367     this.getEncodedHex = function() {
368 	if (this.asn1NotBefore == null || this.asn1NotAfter == null)
369 	    throw "notBefore and/or notAfter not set";
370 	var asn1Validity = 
371 	    new KJUR.asn1.DERSequence({'array':[this.asn1NotBefore, this.asn1NotAfter]});

373 	this.asn1Array = new Array();
374 
375 	this.asn1Array.push(this.asn1Version);
376 	this.asn1Array.push(this.asn1SerialNumber);
377 	this.asn1Array.push(this.asn1SignatureAlg);

382

386 							   'tag': 'a3',
387 							   'obj': extSeq});
388 	    this.asn1Array.push(extTagObj);
389 	}
390 
391 	var o = new KJUR.asn1.DERSequence({"array": this.asn1Array});
392 	this.hTLV = o.getEncodedHex();
393 	this.isModified = false;
394 	return this.hTLV;
395     };
396 
397     this._initialize();
398 };

405 /**
406  * base Extension ASN.1 structure class
407  * @name KJUR.asn1.x509.Extension
408  * @class base Extension ASN.1 structure class
409  * @param {Array} params associative array of parameters (ex. {'critical': true})
410  * @extends KJUR.asn1.ASN1Object
411  * @description
412  * @example
413  * // Extension  ::=  SEQUENCE  {
414  * //     extnID      OBJECT IDENTIFIER,
415  * //     critical    BOOLEAN DEFAULT FALSE,
416  * //     extnValue   OCTET STRING  }
417  */

419     KJUR.asn1.x509.Extension.superclass.constructor.call(this);
420     var asn1ExtnValue = null;
421 
422     this.getEncodedHex = function() {
423 	var asn1Oid = new KJUR.asn1.DERObjectIdentifier({'oid': this.oid});
424 	var asn1EncapExtnValue = 
425 	    new KJUR.asn1.DEROctetString({'hex': this.getExtnValueHex()});
426 
427 	var asn1Array = new Array();
428 	asn1Array.push(asn1Oid);
429 	if (this.critical) asn1Array.push(new KJUR.asn1.DERBoolean());
430 	asn1Array.push(asn1EncapExtnValue);
431 
432 	var asn1Seq = new KJUR.asn1.DERSequence({'array': asn1Array});
433 	return asn1Seq.getEncodedHex();
434     };
435 
436     this.critical = false;
437     if (typeof params != "undefined") {
438 	if (typeof params['critical'] != "undefined") {
439 	    this.critical = params['critical'];
440 	}
441     }
442 };
