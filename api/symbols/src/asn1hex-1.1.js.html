 23 /*
 24  * MEMO:
 25  *   f('3082025b02...', 2) ... 82025b ... 3bytes
 26  *   f('020100', 2) ... 01 ... 1byte
 27  *   f('0203001...', 2) ... 03 ... 1byte
 28  *   f('02818003...', 2) ... 8180 ... 2bytes
 29  *   f('3080....0000', 2) ... 80 ... -1
 30  *
 31  *   Requirements:
 32  *   - ASN.1 type octet length MUST be 1. 
 33  *     (i.e. ASN.1 primitives like SET, SEQUENCE, INTEGER, OCTETSTRING ...)
 34  */

 36 /**
 37  * ASN.1 DER encoded hexadecimal string utility class
 38  * @name ASN1HEX
 39  * @class ASN.1 DER encoded hexadecimal string utility class
 40  * @since jsrsasign 1.1
 41  * @description
 42  * This class provides a parser for hexadecimal string of
 43  * DER encoded ASN.1 binary data.
 44  * Here are major methods of this class.
 45  * <ul>
 46  * <li><b>ACCESS BY POSITION</b>
 47  *   <ul>
 48  *   <li>{@link ASN1HEX.getTLV} - get ASN.1 TLV at specified position</li>
 49  *   <li>{@link ASN1HEX.getV} - get ASN.1 V at specified position</li>
 50  *   <li>{@link ASN1HEX.getVlen} - get integer ASN.1 L at specified position</li>
 51  *   <li>{@link ASN1HEX.getVidx} - get ASN.1 V position from its ASN.1 TLV position</li>
 52  *   <li>{@link ASN1HEX.getL} - get hexadecimal ASN.1 L at specified position</li>
 53  *   <li>{@link ASN1HEX.getLblen} - get byte length for ASN.1 L(length) bytes</li>
 54  *   <li>(DEPRECATED) {@link ASN1HEX.getHexOfTLV_AtObj} - get ASN.1 TLV at specified position</li>
 55  *   <li>(DEPRECATED) {@link ASN1HEX.getHexOfV_AtObj} - get ASN.1 V at specified position</li>
 56  *   <li>(DEPRECATED) {@link ASN1HEX.getHexOfL_AtObj} - get hexadecimal ASN.1 L at specified position</li>
 57  *   <li>(DEPRECATED) {@link ASN1HEX.getIntOfL_AtObj} - get integer ASN.1 L at specified position</li>
 58  *   <li>(DEPRECATED) {@link ASN1HEX.getStartPosOfV_AtObj} - get ASN.1 V position from its ASN.1 TLV position</li>
 59  *   <li>(DEPRECATED) {@link ASN1HEX.getByteLengthOfL_AtObj} - get byte length for ASN.1 L(length) bytes</li>
 60  *   </ul>
 61  * </li>
 62  * <li><b>ACCESS FOR CHILD ITEM</b>
 63  *   <ul>
 64  *   <li>{@link ASN1HEX.getNthChildIndex_AtObj} - get nth child index at specified position</li>
 65  *   <li>{@link ASN1HEX.getPosArrayOfChildren_AtObj} - get indexes of children</li>
 66  *   <li>{@link ASN1HEX.getPosOfNextSibling_AtObj} - get position of next sibling</li>
 67  *   </ul>
 68  * </li>
 69  * <li><b>ACCESS NESTED ASN.1 STRUCTURE</b>
 70  *   <ul>
 71  *   <li>{@link ASN1HEX.getTLVbyList} - get ASN.1 TLV at specified list index</li>
 72  *   <li>{@link ASN1HEX.getVbyList} - get ASN.1 V at specified nth list index with checking expected tag</li>
 73  *   <li>{@link ASN1HEX.getIdxbyList} - get index at specified list index</li>
 74  *   <li>(DEPRECATED) {@link ASN1HEX.getDecendantHexTLVByNthList} - get ASN.1 TLV at specified list index</li>
 75  *   <li>(DEPRECATED) {@link ASN1HEX.getDecendantHexVByNthList} - get ASN.1 V at specified list index</li>
 76  *   <li>(DEPRECATED) {@link ASN1HEX.getDecendantIndexByNthList} - get index at specified list index</li>
 77  *   </ul>
 78  * </li>
 79  * <li><b>UTILITIES</b>
 80  *   <ul>
 81  *   <li>{@link ASN1HEX.dump} - dump ASN.1 structure</li>
 82  *   <li>{@link ASN1HEX.isASN1HEX} - check whether ASN.1 hexadecimal string or not</li>
 83  *   <li>{@link ASN1HEX.hextooidstr} - convert hexadecimal string of OID to dotted integer list</li>
 84  *   </ul>
 85  * </li>
 86  * </ul>
 87  */

 91 /**
 92  * get byte length for ASN.1 L(length) bytes<br/>
 93  * @name getLblen
 94  * @memberOf ASN1HEX
 95  * @function
 96  * @param {String} s hexadecimal string of ASN.1 DER encoded data
 97  * @param {Number} idx string index
 98  * @return byte length for ASN.1 L(length) bytes
 99  * @since jsrsasign 7.2.0 asn1hex 1.1.11
100  * @example
101  * ASN1HEX.getLblen('020100', 0) → 1 for '01'
102  * ASN1HEX.getLblen('020200', 0) → 1 for '02'
103  * ASN1HEX.getLblen('02818003...', 0) → 2 for '8180'
104  * ASN1HEX.getLblen('0282025b03...', 0) → 3 for '82025b'
105  * ASN1HEX.getLblen('0280020100...', 0) → -1 for '80' BER indefinite length
106  * ASN1HEX.getLblen('02ffab...', 0) → -2 for malformed ASN.1 length
107  */
108 ASN1HEX.getLblen = function(s, idx) {
109     if (s.substr(idx + 2, 1) != '8') return 1;
110     var i = parseInt(s.substr(idx + 3, 1));
111     if (i == 0) return -1;             // length octet '80' indefinite length
112     if (0 < i && i < 10) return i + 1; // including '8?' octet;
113     return -2;                         // malformed format
114 };

128 /**
129  * get hexadecimal string for ASN.1 L(length) bytes<br/>
130  * @name getL
131  * @memberOf ASN1HEX
132  * @function
133  * @param {String} s hexadecimal string of ASN.1 DER encoded data
134  * @param {Number} idx string index to get L of ASN.1 object
135  * @return {String} hexadecimal string for ASN.1 L(length) bytes
136  * @since jsrsasign 7.2.0 asn1hex 1.1.11
137  */
138 ASN1HEX.getL = function(s, idx) {
139     var len = ASN1HEX.getLblen(s, idx);
140     if (len < 1) return '';
141     return s.substr(idx + 2, len * 2);
142 };

156 /**
157  * get integer value of ASN.1 length for ASN.1 data<br/>
158  * @name getVblen
159  * @memberOf ASN1HEX
160  * @function
161  * @param {String} s hexadecimal string of ASN.1 DER encoded data
162  * @param {Number} idx string index
163  * @return ASN.1 L(length) integer value
164  * @since jsrsasign 7.2.0 asn1hex 1.1.11
165  */
166 /*
167  getting ASN.1 length value at the position 'idx' of
168  hexa decimal string 's'.
169  f('3082025b02...', 0) ... 82025b ... ???
170  f('020100', 0) ... 01 ... 1
171  f('0203001...', 0) ... 03 ... 3
172  f('02818003...', 0) ... 8180 ... 128
173  */
174 ASN1HEX.getVblen = function(s, idx) {
175     var hLen, bi;
176     hLen = ASN1HEX.getL(s, idx);
177     if (hLen == '') return -1;
178     if (hLen.substr(0, 1) === '8') {
179         bi = new BigInteger(hLen.substr(2), 16);
180     } else {
181         bi = new BigInteger(hLen, 16);
182     }
183     return bi.intValue();
184 };

198 /**
199  * get ASN.1 value starting string position for ASN.1 object refered by index 'idx'.
200  * @name getVidx
201  * @memberOf ASN1HEX
202  * @function
203  * @param {String} s hexadecimal string of ASN.1 DER encoded data
204  * @param {Number} idx string index
205  * @since jsrsasign 7.2.0 asn1hex 1.1.11
206  */
207 ASN1HEX.getVidx = function(s, idx) {
208     var l_len = ASN1HEX.getLblen(s, idx);
209     if (l_len < 0) return l_len;
210     return idx + (l_len + 1) * 2;
211 };

282  * get next sibling starting index for ASN.1 object string<br/>
283  * @name getNextSiblingIdx
284  * @memberOf ASN1HEX
285  * @function
286  * @param {String} s hexadecimal string of ASN.1 DER encoded data

288  * @return next sibling starting index for ASN.1 object string

316 /**
317  * get array of string indexes of child ASN.1 objects<br/>
318  * @name getChildIdx
319  * @memberOf ASN1HEX
320  * @function
321  * @param {String} h hexadecimal string of ASN.1 DER encoded data
322  * @param {Number} pos start string index of ASN.1 object
323  * @return {Array of Number} array of indexes for childen of ASN.1 objects
324  * @since jsrsasign 7.2.0 asn1hex 1.1.11
325  * @description
326  * This method returns array of integers for a concatination of ASN.1 objects
327  * in a ASN.1 value. As for BITSTRING, one byte of unusedbits is skipped.
328  * As for other ASN.1 simple types such as INTEGER, OCTET STRING or PRINTABLE STRING,
329  * it returns a array of a string index of its ASN.1 value.<br/>
330  * NOTE: Since asn1hex 1.1.7 of jsrsasign 6.1.2, Encapsulated BitString is supported.
331  * @example
332  * ASN1HEX.getChildIdx("0203012345", 0) ⇒ [4] // INTEGER 012345
333  * ASN1HEX.getChildIdx("1303616161", 0) ⇒ [4] // PrintableString aaa
334  * ASN1HEX.getChildIdx("030300ffff", 0) ⇒ [6] // BITSTRING ffff (unusedbits=00a)
335  * ASN1HEX.getChildIdx("3006020104020105", 0) ⇒ [4, 10] // SEQUENCE(INT4,INT5)
336  */
337 ASN1HEX.getChildIdx = function(h, pos) {
338     var _ASN1HEX = ASN1HEX;
339     var a = new Array();
340     var p0 = _ASN1HEX.getVidx(h, pos);
341     if (h.substr(pos, 2) == "03") {
342 	a.push(p0 + 2); // BITSTRING value without unusedbits
343     } else {
344 	a.push(p0);
345     }
346 
347     var blen = _ASN1HEX.getVblen(h, pos);
348     var p = p0;
349     var k = 0;
350     while (1) {
351         var pNext = _ASN1HEX.getNextSiblingIdx(h, p);
352         if (pNext == null || (pNext - p0  >= (blen * 2))) break;
353         if (k >= 200) break;
354             
355         a.push(pNext);
356         p = pNext;
357             
358         k++;
359     }
360     
361     return a;
362 };

411 // ========== decendant methods ==============================
412 /**
413  * get string index of nth child object of ASN.1 object refered by h, idx<br/>
414  * @name getIdxbyList
415  * @memberOf ASN1HEX
416  * @function
417  * @param {String} h hexadecimal string of ASN.1 DER encoded data
418  * @param {Number} currentIndex start string index of ASN.1 object
419  * @param {Array of Number} nthList array list of nth
420  * @param {String} checkingTag (OPTIONAL) string of expected ASN.1 tag for nthList 
421  * @return {Number} string index refered by nthList
422  * @since jsrsasign 7.1.4 asn1hex 1.1.10.
423  * @description
424  * @example
425  * The "nthList" is a index list of structured ASN.1 object
426  * reference. Here is a sample structure and "nthList"s which
427  * refers each objects.
428  *
429  * SQUENCE               - 
430  *   SEQUENCE            - [0]
431  *     IA5STRING 000     - [0, 0]
432  *     UTF8STRING 001    - [0, 1]
433  *   SET                 - [1]
434  *     IA5STRING 010     - [1, 0]
435  *     UTF8STRING 011    - [1, 1]
436  */
437 ASN1HEX.getIdxbyList = function(h, currentIndex, nthList, checkingTag) {
438     var _ASN1HEX = ASN1HEX;
439     var firstNth, a;
440     if (nthList.length == 0) {
441 	if (checkingTag !== undefined) {
442             if (h.substr(currentIndex, 2) !== checkingTag) {
443 		throw "checking tag doesn't match: " + 
444                     h.substr(currentIndex, 2) + "!=" + checkingTag;
445             }
446 	}
447         return currentIndex;
448     }
449     firstNth = nthList.shift();
450     a = _ASN1HEX.getChildIdx(h, currentIndex);
451     return _ASN1HEX.getIdxbyList(h, a[firstNth], nthList, checkingTag);
452 };

472 /**
473  * get ASN.1 TLV by nthList<br/>
474  * @name getTLVbyList
475  * @memberOf ASN1HEX
476  * @function
477  * @param {String} h hexadecimal string of ASN.1 structure
478  * @param {Integer} currentIndex string index to start searching in hexadecimal string "h"
479  * @param {Array} nthList array of nth list index
480  * @param {String} checkingTag (OPTIONAL) string of expected ASN.1 tag for nthList 
481  * @since jsrsasign 7.1.4 asn1hex 1.1.10
482  * @description
483  * This static method is to get a ASN.1 value which specified "nthList" position
484  * with checking expected tag "checkingTag".
485  */
486 ASN1HEX.getTLVbyList = function(h, currentIndex, nthList, checkingTag) {
487     var _ASN1HEX = ASN1HEX;
488     var idx = _ASN1HEX.getIdxbyList(h, currentIndex, nthList);
489     if (idx === undefined) {
490         throw "can't find nthList object";

495                 h.substr(idx,2) + "!=" + checkingTag;
496         }
497     }
498     return _ASN1HEX.getTLV(h, idx);
499 };

558 /**
559  * get OID string from hexadecimal encoded value<br/>
560  * @name hextooidstr
561  * @memberOf ASN1HEX
562  * @function
563  * @param {String} hex hexadecmal string of ASN.1 DER encoded OID value
564  * @return {String} OID string (ex. '1.2.3.4.567')
565  * @since asn1hex 1.1.5
566  */
567 ASN1HEX.hextooidstr = function(hex) {
568     var zeroPadding = function(s, len) {
569         if (s.length >= len) return s;
570         return new Array(len - s.length + 1).join('0') + s;
571     };
572 
573     var a = [];
574 
575     // a[0], a[1]
576     var hex0 = hex.substr(0, 2);
577     var i0 = parseInt(hex0, 16);
578     a[0] = new String(Math.floor(i0 / 40));
579     a[1] = new String(i0 % 40);
580 
581     // a[2]..a[n]
582    var hex1 = hex.substr(2);
583     var b = [];
584     for (var i = 0; i < hex1.length / 2; i++) {
585     b.push(parseInt(hex1.substr(i * 2, 2), 16));
586     }
587     var c = [];
588     var cbin = "";
589     for (var i = 0; i < b.length; i++) {
590         if (b[i] & 0x80) {
591             cbin = cbin + zeroPadding((b[i] & 0x7f).toString(2), 7);
592         } else {
593             cbin = cbin + zeroPadding((b[i] & 0x7f).toString(2), 7);
594             c.push(new String(parseInt(cbin, 2)));
595             cbin = "";
596         }
597     }
598 
599     var s = a.join(".");
600     if (c.length > 0) s = s + "." + c.join(".");
601     return s;
602 };

604 /**
605  * get string of simple ASN.1 dump from hexadecimal ASN.1 data<br/>
606  * @name dump
607  * @memberOf ASN1HEX
608  * @function
609  * @param {Object} hexOrObj hexadecmal string of ASN.1 data or ASN1Object object
610  * @param {Array} flags associative array of flags for dump (OPTION)
611  * @param {Number} idx string index for starting dump (OPTION)
612  * @param {String} indent indent string (OPTION)
613  * @return {String} string of simple ASN.1 dump
614  * @since jsrsasign 4.8.3 asn1hex 1.1.6
615  * @description
616  * This method will get an ASN.1 dump from
617  * hexadecmal string of ASN.1 DER encoded data.
618  * Here are features:
619  * <ul>
620  * <li>ommit long hexadecimal string</li>
621  * <li>dump encapsulated OCTET STRING (good for X.509v3 extensions)</li>
622  * <li>structured/primitive context specific tag support (i.e. [0], [3] ...)</li>
623  * <li>automatic decode for implicit primitive context specific tag 
624  * (good for X.509v3 extension value)
625  *   <ul>
626  *   <li>if hex starts '68747470'(i.e. http) it is decoded as utf8 encoded string.</li>
627  *   <li>if it is in 'subjectAltName' extension value and is '[2]'(dNSName) tag
628  *   value will be encoded as utf8 string</li>
629  *   <li>otherwise it shows as hexadecimal string</li>
630  *   </ul>
631  * </li>
632  * </ul>
633  * NOTE1: Argument {@link KJUR.asn1.ASN1Object} object is supported since
634  * jsrsasign 6.2.4 asn1hex 1.0.8
635  * @example
636  * // 1) ASN.1 INTEGER
637  * ASN1HEX.dump('0203012345')
638  * ↓
639  * INTEGER 012345
640  *
641  * // 2) ASN.1 Object Identifier
642  * ASN1HEX.dump('06052b0e03021a')
643  * ↓
644  * ObjectIdentifier sha1 (1 3 14 3 2 26)
645  *
646  * // 3) ASN.1 SEQUENCE
647  * ASN1HEX.dump('3006020101020102')
648  * ↓
649  * SEQUENCE
650  *   INTEGER 01
651  *   INTEGER 02
652  *
653  * // 4) ASN.1 SEQUENCE since jsrsasign 6.2.4
654  * o = KJUR.asn1.ASN1Util.newObject({seq: [{int: 1}, {int: 2}]});
655  * ASN1HEX.dump(o)
656  * ↓
657  * SEQUENCE
658  *   INTEGER 01
659  *   INTEGER 02
660  * // 5) ASN.1 DUMP FOR X.509 CERTIFICATE
661  * ASN1HEX.dump(ASN1HEX.pemToHex(certPEM))
662  * ↓
663  * SEQUENCE
664  *   SEQUENCE
665  *     [0]
666  *       INTEGER 02
667  *     INTEGER 0c009310d206dbe337553580118ddc87
668  *     SEQUENCE
669  *       ObjectIdentifier SHA256withRSA (1 2 840 113549 1 1 11)
670  *       NULL
671  *     SEQUENCE
672  *       SET
673  *         SEQUENCE
674  *           ObjectIdentifier countryName (2 5 4 6)
675  *           PrintableString 'US'
676  *             :
677  */
678 ASN1HEX.dump = function(hexOrObj, flags, idx, indent) {
679     var _ASN1HEX = ASN1HEX;
680     var _getV = _ASN1HEX.getV;
681     var _dump = _ASN1HEX.dump;
682     var _getChildIdx = _ASN1HEX.getChildIdx;
683 
684     var hex = hexOrObj;
685     if (hexOrObj instanceof KJUR.asn1.ASN1Object)
686 	hex = hexOrObj.getEncodedHex();
687 
688     var _skipLongHex = function(hex, limitNumOctet) {
689 	if (hex.length <= limitNumOctet * 2) {
690 	    return hex;
691 	} else {
692 	    var s = hex.substr(0, limitNumOctet) + 
693 		    "..(total " + hex.length / 2 + "bytes).." +
694 		    hex.substr(hex.length - limitNumOctet, limitNumOctet);
695 	    return s;
696 	};
697     };
698 
699     if (flags === undefined) flags = { "ommit_long_octet": 32 };
700     if (idx === undefined) idx = 0;
701     if (indent === undefined) indent = "";
702     var skipLongHex = flags.ommit_long_octet;
703 
704     if (hex.substr(idx, 2) == "01") {
705 	var v = _getV(hex, idx);
706 	if (v == "00") {
707 	    return indent + "BOOLEAN FALSE\n";
708 	} else {
709 	    return indent + "BOOLEAN TRUE\n";
710 	}
711     }
712     if (hex.substr(idx, 2) == "02") {
713 	var v = _getV(hex, idx);
714 	return indent + "INTEGER " + _skipLongHex(v, skipLongHex) + "\n";
715     }
716     if (hex.substr(idx, 2) == "03") {
717 	var v = _getV(hex, idx);
718 	return indent + "BITSTRING " + _skipLongHex(v, skipLongHex) + "\n";
719     }
720     if (hex.substr(idx, 2) == "04") {
721 	var v = _getV(hex, idx);
722 	if (_ASN1HEX.isASN1HEX(v)) {
723 	    var s = indent + "OCTETSTRING, encapsulates\n";
724 	    s = s + _dump(v, flags, 0, indent + "  ");
725 	    return s;
726 	} else {
727 	    return indent + "OCTETSTRING " + _skipLongHex(v, skipLongHex) + "\n";
728 	}
729     }
730     if (hex.substr(idx, 2) == "05") {
731 	return indent + "NULL\n";
732     }
733     if (hex.substr(idx, 2) == "06") {
734 	var hV = _getV(hex, idx);
735         var oidDot = KJUR.asn1.ASN1Util.oidHexToInt(hV);
736         var oidName = KJUR.asn1.x509.OID.oid2name(oidDot);
737 	var oidSpc = oidDot.replace(/\./g, ' ');
738         if (oidName != '') {
739   	    return indent + "ObjectIdentifier " + oidName + " (" + oidSpc + ")\n";
740 	} else {
741   	    return indent + "ObjectIdentifier (" + oidSpc + ")\n";
742 	}
743     }
744     if (hex.substr(idx, 2) == "0c") {
745 	return indent + "UTF8String '" + hextoutf8(_getV(hex, idx)) + "'\n";
746     }
747     if (hex.substr(idx, 2) == "13") {
748 	return indent + "PrintableString '" + hextoutf8(_getV(hex, idx)) + "'\n";
749     }
750     if (hex.substr(idx, 2) == "14") {
751 	return indent + "TeletexString '" + hextoutf8(_getV(hex, idx)) + "'\n";
752     }
753     if (hex.substr(idx, 2) == "16") {
754 	return indent + "IA5String '" + hextoutf8(_getV(hex, idx)) + "'\n";
755     }
756     if (hex.substr(idx, 2) == "17") {
757 	return indent + "UTCTime " + hextoutf8(_getV(hex, idx)) + "\n";
758     }
759     if (hex.substr(idx, 2) == "18") {
760 	return indent + "GeneralizedTime " + hextoutf8(_getV(hex, idx)) + "\n";
761     }
762     if (hex.substr(idx, 2) == "30") {
763 	if (hex.substr(idx, 4) == "3000") {
764 	    return indent + "SEQUENCE {}\n";
765 	}
766 
767 	var s = indent + "SEQUENCE\n";
768 	var aIdx = _getChildIdx(hex, idx);
769 
770 	var flagsTemp = flags;
771 	
772 	if ((aIdx.length == 2 || aIdx.length == 3) &&
773 	    hex.substr(aIdx[0], 2) == "06" &&
774 	    hex.substr(aIdx[aIdx.length - 1], 2) == "04") { // supposed X.509v3 extension
775 	    var oidName = _ASN1HEX.oidname(_getV(hex, aIdx[0]));
776 	    var flagsClone = JSON.parse(JSON.stringify(flags));
777 	    flagsClone.x509ExtName = oidName;
778 	    flagsTemp = flagsClone;
779 	}
780 	
781 	for (var i = 0; i < aIdx.length; i++) {
782 	    s = s + _dump(hex, flagsTemp, aIdx[i], indent + "  ");
783 	}
784 	return s;
785     }
786     if (hex.substr(idx, 2) == "31") {
787 	var s = indent + "SET\n";
788 	var aIdx = _getChildIdx(hex, idx);
789 	for (var i = 0; i < aIdx.length; i++) {
790 	    s = s + _dump(hex, flags, aIdx[i], indent + "  ");
791 	}
792 	return s;
793     }
794     var tag = parseInt(hex.substr(idx, 2), 16);
795     if ((tag & 128) != 0) { // context specific 
796 	var tagNumber = tag & 31;
797 	if ((tag & 32) != 0) { // structured tag
798 	    var s = indent + "[" + tagNumber + "]\n";
799 	    var aIdx = _getChildIdx(hex, idx);
800 	    for (var i = 0; i < aIdx.length; i++) {
801 		s = s + _dump(hex, flags, aIdx[i], indent + "  ");
802 	    }
803 	    return s;
804 	} else { // primitive tag
805 	    var v = _getV(hex, idx);
806 	    if (v.substr(0, 8) == "68747470") { // http
807 		v = hextoutf8(v);
808 	    }
809 	    if (flags.x509ExtName === "subjectAltName" &&
810 		tagNumber == 2) {
811 		v = hextoutf8(v);
812 	    }
813 	    
814 	    var s = indent + "[" + tagNumber + "] " + v + "\n";
815 	    return s;
816 	}
817     }
818     return indent + "UNKNOWN(" + hex.substr(idx, 2) + ") " + 
819 	   _getV(hex, idx) + "\n";
820 };

822 /**
823  * check wheather the string is ASN.1 hexadecimal string or not
824  * @name isASN1HEX
825  * @memberOf ASN1HEX
826  * @function
827  * @param {String} hex string to check whether it is hexadecmal string for ASN.1 DER or not
828  * @return {Boolean} true if it is hexadecimal string of ASN.1 data otherwise false
829  * @since jsrsasign 4.8.3 asn1hex 1.1.6
830  * @description
831  * This method checks wheather the argument 'hex' is a hexadecimal string of
832  * ASN.1 data or not.
833  * @example
834  * ASN1HEX.isASN1HEX('0203012345') → true // PROPER ASN.1 INTEGER
835  * ASN1HEX.isASN1HEX('0203012345ff') → false // TOO LONG VALUE
836  * ASN1HEX.isASN1HEX('02030123') → false // TOO SHORT VALUE
837  * ASN1HEX.isASN1HEX('fa3bcd') → false // WRONG FOR ASN.1
838  */
839 ASN1HEX.isASN1HEX = function(hex) {
840     var _ASN1HEX = ASN1HEX;
841     if (hex.length % 2 == 1) return false;
842 
843     var intL = _ASN1HEX.getVblen(hex, 0);
844     var tV = hex.substr(0, 2);
845     var lV = _ASN1HEX.getL(hex, 0);
846     var hVLength = hex.length - tV.length - lV.length;
847     if (hVLength == intL * 2) return true;
848 
849     return false;
850 };

878 /**
879  * get hexacedimal string from PEM format data<br/>
880  * @name pemToHex
881  * @memberOf ASN1HEX
882  * @function
883  * @param {String} s PEM formatted string
884  * @param {String} sHead PEM header string without BEGIN/END(OPTION)
885  * @return {String} hexadecimal string data of PEM contents
886  * @since jsrsasign 7.0.1 asn1hex 1.1.9
887  * @description
888  * This static method gets a hexacedimal string of contents 
889  * from PEM format data. You can explicitly specify PEM header 
890  * by sHead argument. 
891  * Any space characters such as white space or new line
892  * will be omitted.<br/>
893  * NOTE: Now {@link KEYUTIL.getHexFromPEM} and {@link X509.pemToHex}
894  * have been deprecated since jsrsasign 7.0.1. 
895  * Please use this method instead.
896  * @example
897  * ASN1HEX.pemToHex("-----BEGIN PUBLIC KEY...") → "3082..."
898  * ASN1HEX.pemToHex("-----BEGIN CERTIFICATE...", "CERTIFICATE") → "3082..."
899  * ASN1HEX.pemToHex(" \r\n-----BEGIN DSA PRIVATE KEY...") → "3082..."
900  */
901 ASN1HEX.pemToHex = function(s, sHead) {
902     if (s.indexOf("-----BEGIN ") == -1)
903         throw "can't find PEM header: " + sHead;
904 
905     if (sHead !== undefined) {
906         s = s.replace("-----BEGIN " + sHead + "-----", "");
907         s = s.replace("-----END " + sHead + "-----", "");
908     } else {
909         s = s.replace(/-----BEGIN [^-]+-----/, '');
910         s = s.replace(/-----END [^-]+-----/, '');
911     }
912     var sB64 = s.replace(/\s+/g, '');
913     var dataHex = b64tohex(sB64);
914     return dataHex;
915 };
