 22 /**
 23  * @name PKCS5PKEY
 24  * @class class for PKCS#5 and PKCS#8 private key 
 25  * @description 
 26  * <br/>
 27  * {@link PKCS5PKEY} class has following features:
 28  * <ul>
 29  * <li>read and parse PEM formatted encrypted PKCS#5 private key
 30  * <li>generate PEM formatted encrypted PKCS#5 private key
 31  * <li>read and parse PEM formatted plain PKCS#8 private key
 32  * <li>read and parse PEM formatted encrypted PKCS#8 private key by PBKDF2/HmacSHA1/3DES
 33  * </ul>
 34  * Currently supports only RSA private key and
 35  * following symmetric key algorithms to protect private key.
 36  * <ul>
 37  * <li>DES-EDE3-CBC</li>
 38  * <li>AES-256-CBC</li>
 39  * <li>AES-192-CBC</li>
 40  * <li>AES-128-CBC</li>
 41  * </ul>
 42  * 
 43  * @example
 44  * Here is an example of PEM formatted encrypted PKCS#5 private key.
 45  * -----BEGIN RSA PRIVATE KEY-----
 46  * Proc-Type: 4,ENCRYPTED
 47  * DEK-Info: AES-256-CBC,40555967F759530864FE022E257DE34E
 48  *
 49  * jV7uXajRw4cccDaliagcqiLOiQEUCe19l761pXRxzgQP+DH4rCi12T4puTdZyy6l
 50  *          ...(snip)...
 51  * qxLS+BASmyGm4DME6m+kltZ12LXwPgNU6+d+XQ4NXSA=
 52  *-----END RSA PRIVATE KEY-----
 53  */

 58     // shared key decryption ------------------------------------------
 59     var decryptAES = function(dataHex, keyHex, ivHex) {
 60 	return decryptGeneral(CryptoJS.AES, dataHex, keyHex, ivHex);
 61     };
 62 
 63     var decrypt3DES = function(dataHex, keyHex, ivHex) {
 64 	return decryptGeneral(CryptoJS.TripleDES, dataHex, keyHex, ivHex);
 65     };

 67     var decryptGeneral = function(f, dataHex, keyHex, ivHex) {
 68 	var data = CryptoJS.enc.Hex.parse(dataHex);
 69 	var key = CryptoJS.enc.Hex.parse(keyHex);
 70 	var iv = CryptoJS.enc.Hex.parse(ivHex);
 71 	var encrypted = {};
 72 	encrypted.key = key;
 73 	encrypted.iv = iv;
 74 	encrypted.ciphertext = data;
 75 	var decrypted = f.decrypt(encrypted, key, { iv: iv });
 76 	return CryptoJS.enc.Hex.stringify(decrypted);
 77     };
 78 
 79     // shared key decryption ------------------------------------------
 80     var encryptAES = function(dataHex, keyHex, ivHex) {
 81 	return encryptGeneral(CryptoJS.AES, dataHex, keyHex, ivHex);
 82     };
 83 
 84     var encrypt3DES = function(dataHex, keyHex, ivHex) {
 85 	return encryptGeneral(CryptoJS.TripleDES, dataHex, keyHex, ivHex);
 86     };

 88     var encryptGeneral = function(f, dataHex, keyHex, ivHex) {
 89 	var data = CryptoJS.enc.Hex.parse(dataHex);
 90 	var key = CryptoJS.enc.Hex.parse(keyHex);
 91 	var iv = CryptoJS.enc.Hex.parse(ivHex);
 92 	var msg = {};
 93 	var encryptedHex = f.encrypt(data, key, { iv: iv });
 94         var encryptedWA = CryptoJS.enc.Hex.parse(encryptedHex.toString());
 95         var encryptedB64 = CryptoJS.enc.Base64.stringify(encryptedWA);
 96 	return encryptedB64;
 97     };

111     var _generateIvSaltHex = function(numBytes) {
112 	var wa = CryptoJS.lib.WordArray.random(numBytes);

117     var _parsePKCS5PEM = function(sPKCS5PEM) {
118 	var info = {};
119 	if (sPKCS5PEM.match(new RegExp("DEK-Info: ([^,]+),([0-9A-Fa-f]+)", "m"))) {
120 	    info.cipher = RegExp.$1;
121 	    info.ivsalt = RegExp.$2;
122 	}
123 	if (sPKCS5PEM.match(new RegExp("-----BEGIN ([A-Z]+) PRIVATE KEY-----"))) {
124 	    info.type = RegExp.$1;
125 	}
126 	var i1 = -1;
127 	var lenNEWLINE = 0;
128 	if (sPKCS5PEM.indexOf("\r\n\r\n") != -1) {
129 	    i1 = sPKCS5PEM.indexOf("\r\n\r\n");
130 	    lenNEWLINE = 2;
131 	}
132 	if (sPKCS5PEM.indexOf("\n\n") != -1) {
133 	    i1 = sPKCS5PEM.indexOf("\n\n");
134 	    lenNEWLINE = 1;
135 	}
136 	var i2 = sPKCS5PEM.indexOf("-----END");
137 	if (i1 != -1 && i2 != -1) {
138 	    var s = sPKCS5PEM.substring(i1 + lenNEWLINE * 2, i2 - lenNEWLINE);
139 	    s = s.replace(/\s+/g, '');
140 	    info.data = s;
141 	}
142 	return info;
143     };

145     var _getKeyAndUnusedIvByPasscodeAndIvsalt = function(algName, passcode, ivsaltHex) {
146 	//alert("ivsaltHex(2) = " + ivsaltHex);
147 	var saltHex = ivsaltHex.substring(0, 16);
148 	//alert("salt = " + saltHex);
149 	    
150 	var salt = CryptoJS.enc.Hex.parse(saltHex);
151 	var data = CryptoJS.enc.Utf8.parse(passcode);
152 	//alert("salt = " + salt);
153 	//alert("data = " + data);
154 
155 	var nRequiredBytes = ALGLIST[algName]['keylen'] + ALGLIST[algName]['ivlen'];
156 	var hHexValueJoined = '';
157 	var hLastValue = null;
158 	//alert("nRequiredBytes = " + nRequiredBytes);
159 	for (;;) {
160 	    var h = CryptoJS.algo.MD5.create();
161 	    if (hLastValue != null) {
162 		h.update(hLastValue);
163 	    }
164 	    h.update(data);
165 	    h.update(salt);
166 	    hLastValue = h.finalize();
167 	    hHexValueJoined = hHexValueJoined + CryptoJS.enc.Hex.stringify(hLastValue);
168 	    //alert("joined = " + hHexValueJoined);
169 	    if (hHexValueJoined.length >= nRequiredBytes * 2) {
170 		break;
171 	    }
172 	}
173 	var result = {};
174 	result.keyhex = hHexValueJoined.substr(0, ALGLIST[algName]['keylen'] * 2);
175 	result.ivhex = hHexValueJoined.substr(ALGLIST[algName]['keylen'] * 2, ALGLIST[algName]['ivlen'] * 2);
176 	return result;
177     };

179     /*
180      * @param {String} privateKeyB64 base64 string of encrypted private key
181      * @param {String} sharedKeyAlgName algorithm name of shared key encryption
182      * @param {String} sharedKeyHex hexadecimal string of shared key to encrypt
183      * @param {String} ivsaltHex hexadecimal string of IV and salt
184      * @param {String} hexadecimal string of decrypted private key

186     var _decryptKeyB64 = function(privateKeyB64, sharedKeyAlgName, sharedKeyHex, ivsaltHex) {
187 	var privateKeyWA = CryptoJS.enc.Base64.parse(privateKeyB64);
188 	var privateKeyHex = CryptoJS.enc.Hex.stringify(privateKeyWA);
189 	var f = ALGLIST[sharedKeyAlgName]['proc'];
190 	var decryptedKeyHex = f(privateKeyHex, sharedKeyHex, ivsaltHex);
191 	return decryptedKeyHex;
192     };

202 	var f = ALGLIST[sharedKeyAlgName]['eproc'];
203 	var encryptedKeyB64 = f(privateKeyHex, sharedKeyHex, ivsaltHex);
204 	return encryptedKeyB64;
205     };

221 	    var s = sPEM;

331          * get PEM formatted encrypted PKCS#5 private key from hexadecimal string of plain private key
332 	 * @name getEryptedPKCS5PEMFromPrvKeyHex
333 	 * @memberOf PKCS5PKEY
334 	 * @function
335 	 * @param {String} hPrvKey hexadecimal string of plain private key
336 	 * @param {String} passcode pass code to protect private key (ex. password)
337 	 * @param {String} sharedKeyAlgName algorithm name to protect private key (ex. AES-256-CBC)
338 	 * @param {String} ivsaltHex hexadecimal string of IV and salt
339 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
340          * @since pkcs5pkey 1.0.2
341 	 * @description
342 	 * <br/>
343 	 * generate PEM formatted encrypted PKCS#5 private key by hexadecimal string encoded
344 	 * ASN.1 object of plain RSA private key.
345 	 * Following arguments can be omitted.
346 	 * <ul>
347 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
348 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
349 	 * </ul>
350 	 * @example
351 	 * var pem = 
352          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password");
353 	 * var pem2 = 
354          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC");
355 	 * var pem3 = 
356          *   PKCS5PKEY.getEryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC", "1f3d02...");
357 	 */
358 	getEryptedPKCS5PEMFromPrvKeyHex: function(hPrvKey, passcode, sharedKeyAlgName, ivsaltHex) {
359 	    var sPEM = "";
360 
361 	    // 1. set sharedKeyAlgName if undefined (default AES-256-CBC)
362 	    if (typeof sharedKeyAlgName == "undefined" || sharedKeyAlgName == null) {
363 		sharedKeyAlgName = "AES-256-CBC";
364 	    }
365 	    if (typeof ALGLIST[sharedKeyAlgName] == "undefined")
366 		throw "PKCS5PKEY unsupported algorithm: " + sharedKeyAlgName;
367 
368 	    // 2. set ivsaltHex if undefined
369 	    if (typeof ivsaltHex == "undefined" || ivsaltHex == null) {
370 		var ivlen = ALGLIST[sharedKeyAlgName]['ivlen'];
371 		var randIV = _generateIvSaltHex(ivlen);
372 		ivsaltHex = randIV.toUpperCase();
373 	    }
374 
375 	    // 3. get shared key
376             //alert("ivsalthex=" + ivsaltHex);
377 	    var sharedKeyInfo = _getKeyAndUnusedIvByPasscodeAndIvsalt(sharedKeyAlgName, passcode, ivsaltHex);
378 	    var sharedKeyHex = sharedKeyInfo.keyhex;
379 	    // alert("sharedKeyHex = " + sharedKeyHex);
380 
381             // 3. get encrypted Key in Base64
382             var encryptedKeyB64 = _encryptKeyHex(hPrvKey, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
383 
384 	    var pemBody = encryptedKeyB64.replace(/(.{64})/g, "$1\r\n");
385 	    var sPEM = "-----BEGIN RSA PRIVATE KEY-----\r\n";
386 	    sPEM += "Proc-Type: 4,ENCRYPTED\r\n";
387 	    sPEM += "DEK-Info: " + sharedKeyAlgName + "," + ivsaltHex + "\r\n";
388 	    sPEM += "\r\n";
389 	    sPEM += pemBody;
390 	    sPEM += "\r\n-----END RSA PRIVATE KEY-----\r\n";
391 
392 	    return sPEM;
393         },

395 	/**
396          * get PEM formatted encrypted PKCS#5 private key from RSAKey object of private key
397 	 * @name getEryptedPKCS5PEMFromRSAKey
398 	 * @memberOf PKCS5PKEY
399 	 * @function
400 	 * @param {RSAKey} pKey RSAKey object of private key
401 	 * @param {String} passcode pass code to protect private key (ex. password)
402 	 * @param {String} alg algorithm name to protect private key (default AES-256-CBC)
403 	 * @param {String} ivsaltHex hexadecimal string of IV and salt (default generated random IV)
404 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
405          * @since pkcs5pkey 1.0.2
406 	 * @description
407 	 * <br/>
408 	 * generate PEM formatted encrypted PKCS#5 private key by
409 	 * {@link RSAKey} object of RSA private key and passcode.
410 	 * Following argument can be omitted.
411 	 * <ul>
412 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
413 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
414 	 * </ul>
415 	 * @example
416 	 * var pkey = new RSAKey();
417 	 * pkey.generate(1024, '10001'); // generate 1024bit RSA private key with public exponent 'x010001'
418 	 * var pem = PKCS5PKEY.getEryptedPKCS5PEMFromRSAKey(pkey, "password");
419 	 */
420         getEryptedPKCS5PEMFromRSAKey: function(pKey, passcode, alg, ivsaltHex) {
421 	    var version = new KJUR.asn1.DERInteger({'int': 0});
422 	    var n = new KJUR.asn1.DERInteger({'bigint': pKey.n});
423 	    var e = new KJUR.asn1.DERInteger({'int': pKey.e});
424 	    var d = new KJUR.asn1.DERInteger({'bigint': pKey.d});
425 	    var p = new KJUR.asn1.DERInteger({'bigint': pKey.p});
426 	    var q = new KJUR.asn1.DERInteger({'bigint': pKey.q});
427 	    var dmp1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmp1});
428 	    var dmq1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmq1});
429 	    var coeff = new KJUR.asn1.DERInteger({'bigint': pKey.coeff});
430 	    var seq = new KJUR.asn1.DERSequence({'array': [version, n, e, d, p, q, dmp1, dmq1, coeff]});
431 	    var hex = seq.getEncodedHex();
432 	    return this.getEryptedPKCS5PEMFromPrvKeyHex(hex, passcode, alg, ivsaltHex);
433         },

435 	/**
436          * generate RSAKey and PEM formatted encrypted PKCS#5 private key
437 	 * @name newEryptedPKCS5PEM
438 	 * @memberOf PKCS5PKEY
439 	 * @function
440 	 * @param {String} passcode pass code to protect private key (ex. password)
441 	 * @param {Integer} keyLen key bit length of RSA key to be generated. (default 1024)
442 	 * @param {String} hPublicExponent hexadecimal string of public exponent (default 10001)
443 	 * @param {String} alg shared key algorithm to encrypt private key (default AES-258-CBC)
444 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
445          * @since pkcs5pkey 1.0.2
446 	 * @example
447 	 * var pem1 = PKCS5PKEY.newEncryptedPKCS5PEM("password");           // RSA1024bit/10001/AES-256-CBC
448 	 * var pem2 = PKCS5PKEY.newEncryptedPKCS5PEM("password", 512);      // RSA 512bit/10001/AES-256-CBC
449 	 * var pem3 = PKCS5PKEY.newEncryptedPKCS5PEM("password", 512, '3'); // RSA 512bit/    3/AES-256-CBC
450 	 */
451 	newEryptedPKCS5PEM: function(passcode, keyLen, hPublicExponent, alg) {
452 	    if (typeof keyLen == "undefined" || keyLen == null) {
453 		keyLen = 1024;
454 	    }
455 	    if (typeof hPublicExponent == "undefined" || hPublicExponent == null) {
456 		hPublicExponent = '10001';
457 	    }
458 	    var pKey = new RSAKey();
459 	    pKey.generate(keyLen, hPublicExponent);
460 	    var pem = null;
461 	    if (typeof alg == "undefined" || alg == null) {
462 		pem = this.getEncryptedPKCS5PEMFromRSAKey(pkey, passcode);
463 	    } else {
464 		pem = this.getEncryptedPKCS5PEMFromRSAKey(pkey, passcode, alg);
465 	    }
466 	    return pem;
467         },

515 	 * @name parseHexOfEncryptedPKCS8
516 	 * @memberOf PKCS5PKEY
517 	 * @function
518 	 * @param {String} passcode passcode to decrypto private key
519 	 * @return {Array} info associative array of PKCS#8 parameters
520          * @since pkcs5pkey 1.0.3
521 	 * @description
522 	 * The associative array which is returned by this method has following properties:
523 	 * <ul>
524 	 * <li>info.pbkdf2Salt - hexadecimal string of PBKDF2 salt</li>
525 	 * <li>info.pkbdf2Iter - iteration count</li>
526 	 * <li>info.ciphertext - hexadecimal string of encrypted private key</li>
527 	 * <li>info.encryptionSchemeAlg - encryption algorithm name (currently TripleDES only)</li>
528 	 * <li>info.encryptionSchemeIV - initial vector for encryption algorithm</li>
529 	 * </ul>
530 	 * Currently, this method only supports PKCS#5v2.0 with PBES2/PBDKF2 of HmacSHA1 and TripleDES.
531 	 * <ul>
532 	 * <li>keyDerivationFunc = pkcs5PBKDF2 with HmacSHA1</li>
533 	 * <li>encryptionScheme = des-EDE3-CBC(i.e. TripleDES</li>
534 	 * </ul>
535 	 * @example
536 	 * // to convert plain PKCS#5 private key to encrypted PKCS#8 private
537 	 * // key with PBKDF2 with TripleDES
538 	 * % openssl pkcs8 -in plain_p5.pem -topk8 -v2 -des3 -out encrypted_p8.pem
539 	 */
540         parseHexOfEncryptedPKCS8: function(sHEX) {
541             var info = {};
542 	    
543 	    var a0 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, 0);
544 	    if (a0.length != 2)
545 		throw "malformed format: SEQUENCE(0).items != 2: " + a0.length;
546 
547 	    // 1. ciphertext
548 	    info.ciphertext = ASN1HEX.getHexOfV_AtObj(sHEX, a0[1]);
549 
550 	    // 2. pkcs5PBES2
551 	    var a0_0 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, a0[0]); 
552 	    if (a0_0.length != 2)
553 		throw "malformed format: SEQUENCE(0.0).items != 2: " + a0_0.length;
554 
555 	    // 2.1 check if pkcs5PBES2(1 2 840 113549 1 5 13)
556 	    if (ASN1HEX.getHexOfV_AtObj(sHEX, a0_0[0]) != "2a864886f70d01050d")
557 		throw "this only supports pkcs5PBES2";
558 
559 	    // 2.2 pkcs5PBES2 param
560             var a0_0_1 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, a0_0[1]); 
561 	    if (a0_0.length != 2)
562 		throw "malformed format: SEQUENCE(0.0.1).items != 2: " + a0_0_1.length;
563 
564 	    // 2.2.1 encryptionScheme
565 	    var a0_0_1_1 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, a0_0_1[1]); 
566 	    if (a0_0_1_1.length != 2)
567 		throw "malformed format: SEQUENCE(0.0.1.1).items != 2: " + a0_0_1_1.length;
568 	    if (ASN1HEX.getHexOfV_AtObj(sHEX, a0_0_1_1[0]) != "2a864886f70d0307")
569 		throw "this only supports TripleDES";
570 	    info.encryptionSchemeAlg = "TripleDES";
571 
572 	    // 2.2.1.1 IV of encryptionScheme
573 	    info.encryptionSchemeIV = ASN1HEX.getHexOfV_AtObj(sHEX, a0_0_1_1[1]);
574 
575 	    // 2.2.2 keyDerivationFunc
576 	    var a0_0_1_0 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, a0_0_1[0]); 
577 	    if (a0_0_1_0.length != 2)
578 		throw "malformed format: SEQUENCE(0.0.1.0).items != 2: " + a0_0_1_0.length;
579 	    if (ASN1HEX.getHexOfV_AtObj(sHEX, a0_0_1_0[0]) != "2a864886f70d01050c")
580 		throw "this only supports pkcs5PBKDF2";
581 
582 	    // 2.2.2.1 pkcs5PBKDF2 param
583 	    var a0_0_1_0_1 = ASN1HEX.getPosArrayOfChildren_AtObj(sHEX, a0_0_1_0[1]); 
584 	    if (a0_0_1_0_1.length < 2)
585 		throw "malformed format: SEQUENCE(0.0.1.0.1).items < 2: " + a0_0_1_0_1.length;
586 
587 	    // 2.2.2.1.1 PBKDF2 salt
588 	    info.pbkdf2Salt = ASN1HEX.getHexOfV_AtObj(sHEX, a0_0_1_0_1[0]);
589 
590 	    // 2.2.2.1.2 PBKDF2 iter
591 	    var iterNumHex = ASN1HEX.getHexOfV_AtObj(sHEX, a0_0_1_0_1[1]);
592 	    try {
593 		info.pbkdf2Iter = parseInt(iterNumHex, 16);
594 	    } catch(ex) {
595 		throw "malformed format pbkdf2Iter: " + iterNumHex;
596 	    }
597 
598 	    return info;
599 	},

603 	 * @name getPBKDF2KeyHexFromParam
604 	 * @memberOf PKCS5PKEY

607 	 * @param {String} passcode passcode to decrypto private key
608 	 * @return {String} hexadecimal string of PBKDF2 key
609          * @since pkcs5pkey 1.0.3
610 	 * @description
611 	 * As for info, this uses following properties:
612 	 * <ul>
613 	 * <li>info.pbkdf2Salt - hexadecimal string of PBKDF2 salt</li>
614 	 * <li>info.pkbdf2Iter - iteration count</li>
615 	 * </ul>
616 	 * Currently, this method only supports PKCS#5v2.0 with PBES2/PBDKF2 of HmacSHA1 and TripleDES.
617 	 * <ul>
618 	 * <li>keyDerivationFunc = pkcs5PBKDF2 with HmacSHA1</li>
619 	 * <li>encryptionScheme = des-EDE3-CBC(i.e. TripleDES</li>
620 	 * </ul>
621 	 * @example
622 	 * // to convert plain PKCS#5 private key to encrypted PKCS#8 private
623 	 * // key with PBKDF2 with TripleDES
624 	 * % openssl pkcs8 -in plain_p5.pem -topk8 -v2 -des3 -out encrypted_p8.pem

626 	getPBKDF2KeyHexFromParam: function(info, passcode) {
627 	    var pbkdf2SaltWS = CryptoJS.enc.Hex.parse(info.pbkdf2Salt);
628 	    var pbkdf2Iter = info.pbkdf2Iter;
629 	    var pbkdf2KeyWS = CryptoJS.PBKDF2(passcode, 
630 					      pbkdf2SaltWS, 
631 					      { keySize: 192/32, iterations: pbkdf2Iter });
632 	    var pbkdf2KeyHex = CryptoJS.enc.Hex.stringify(pbkdf2KeyWS);
633 	    return pbkdf2KeyHex;
