 23 /**
 24  * @name PKCS5PKEY
 25  * @class class for PKCS#5 and PKCS#8 private key 
 26  * @description 
 27  * <br/>
 28  * {@link PKCS5PKEY} class has following features:
 29  * <ul>
 30  * <li>read and parse PEM formatted encrypted PKCS#5 private key
 31  * <li>generate PEM formatted encrypted PKCS#5 private key
 32  * <li>read and parse PEM formatted plain PKCS#8 private key
 33  * <li>read and parse PEM formatted encrypted PKCS#8 private key by PBKDF2/HmacSHA1/3DES
 34  * </ul>
 35  * Currently supports only RSA private key and
 36  * following symmetric key algorithms to protect private key.
 37  * <ul>
 38  * <li>DES-EDE3-CBC</li>
 39  * <li>AES-256-CBC</li>
 40  * <li>AES-192-CBC</li>
 41  * <li>AES-128-CBC</li>
 42  * </ul>
 43  * 
 44  * <h5>METHOD SUMMARY</h5>
 45  * <dl>
 46  * <dt><b>PKCS8 PRIVATE KEY METHODS</b><dd>
 47  * <ul>
 48  * <li>{@link PKCS5PKEY.getRSAKeyFromPlainPKCS8PEM} - convert plain PKCS8 PEM to RSAKey object</li>
 49  * <li>{@link PKCS5PKEY.getRSAKeyFromPlainPKCS8Hex} - convert plain PKCS8 hexadecimal data to RSAKey object</li>
 50  * <li>{@link PKCS5PKEY.getRSAKeyFromEncryptedPKCS8PEM} - convert encrypted PKCS8 PEM to RSAKey object</li>
 51  * <li>{@link PKCS5PKEY.getPlainPKCS8HexFromEncryptedPKCS8PEM} - convert encrypted PKCS8 PEM to plain PKCS8 Hex</li>
 52  * </ul>
 53  * <dt><b>PKCS5 PRIVATE KEY METHODS</b><dd>
 54  * <ul>
 55  * <li>{@link PKCS5PKEY.getRSAKeyFromEncryptedPKCS5PEM} - convert encrypted PKCS5 PEM to RSAKey object</li>
 56  * <li>{@link PKCS5PKEY.getEncryptedPKCS5PEMFromRSAKey} - convert RSAKey object to encryped PKCS5 PEM</li>
 57  * <li>{@link PKCS5PKEY.newEncryptedPKCS5PEM} - generate RSAKey and its encrypted PKCS5 PEM</li>
 58  * </ul>
 59  * <dt><b>PKCS8 PUBLIC KEY METHODS</b><dd>
 60  * <ul>
 61  * <li>{@link PKCS5PKEY.getKeyFromPublicPKCS8PEM} - convert encrypted PKCS8 PEM to RSAKey/ECDSA object</li>
 62  * <li>{@link PKCS5PKEY.getKeyFromPublicPKCS8Hex} - convert encrypted PKCS8 Hex to RSAKey/ECDSA object</li>
 63  * <li>{@link PKCS5PKEY.getRSAKeyFromPublicPKCS8PEM} - convert encrypted PKCS8 PEM to RSAKey object</li>
 64  * <li>{@link PKCS5PKEY.getRSAKeyFromPublicPKCS8Hex} - convert encrypted PKCS8 Hex to RSAKey object</li>
 65  * </ul>
 66  * <dt><b>UTITILIY METHODS</b><dd>
 67  * <ul>
 68  * <li>{@link PKCS5PKEY.getHexFromPEM} - convert PEM string to hexadecimal data</li>
 69  * <li>{@link PKCS5PKEY.getDecryptedKeyHexByKeyIV} - decrypt key by sharedKey and IV</li>
 70  * </ul>
 71  * </dl>

147     var _parsePKCS5PEM = function(sPKCS5PEM) {
148 	var info = {};
149 	if (sPKCS5PEM.match(new RegExp("DEK-Info: ([^,]+),([0-9A-Fa-f]+)", "m"))) {
150 	    info.cipher = RegExp.$1;
151 	    info.ivsalt = RegExp.$2;
152 	}
153 	if (sPKCS5PEM.match(new RegExp("-----BEGIN ([A-Z]+) PRIVATE KEY-----"))) {
154 	    info.type = RegExp.$1;
155 	}
156 	var i1 = -1;
157 	var lenNEWLINE = 0;
158 	if (sPKCS5PEM.indexOf("\r\n\r\n") != -1) {
159 	    i1 = sPKCS5PEM.indexOf("\r\n\r\n");
160 	    lenNEWLINE = 2;
161 	}
162 	if (sPKCS5PEM.indexOf("\n\n") != -1) {
163 	    i1 = sPKCS5PEM.indexOf("\n\n");
164 	    lenNEWLINE = 1;
165 	}
166 	var i2 = sPKCS5PEM.indexOf("-----END");
167 	if (i1 != -1 && i2 != -1) {
168 	    var s = sPKCS5PEM.substring(i1 + lenNEWLINE * 2, i2 - lenNEWLINE);
169 	    s = s.replace(/\s+/g, '');
170 	    info.data = s;
171 	}
172 	return info;
173     };

175     var _getKeyAndUnusedIvByPasscodeAndIvsalt = function(algName, passcode, ivsaltHex) {
176 	//alert("ivsaltHex(2) = " + ivsaltHex);
177 	var saltHex = ivsaltHex.substring(0, 16);
178 	//alert("salt = " + saltHex);
179 	    
180 	var salt = CryptoJS.enc.Hex.parse(saltHex);
181 	var data = CryptoJS.enc.Utf8.parse(passcode);
182 	//alert("salt = " + salt);
183 	//alert("data = " + data);
184 
185 	var nRequiredBytes = ALGLIST[algName]['keylen'] + ALGLIST[algName]['ivlen'];
186 	var hHexValueJoined = '';
187 	var hLastValue = null;
188 	//alert("nRequiredBytes = " + nRequiredBytes);
189 	for (;;) {
190 	    var h = CryptoJS.algo.MD5.create();
191 	    if (hLastValue != null) {
192 		h.update(hLastValue);
193 	    }
194 	    h.update(data);
195 	    h.update(salt);
196 	    hLastValue = h.finalize();
197 	    hHexValueJoined = hHexValueJoined + CryptoJS.enc.Hex.stringify(hLastValue);
198 	    //alert("joined = " + hHexValueJoined);
199 	    if (hHexValueJoined.length >= nRequiredBytes * 2) {
200 		break;
201 	    }
202 	}
203 	var result = {};
204 	result.keyhex = hHexValueJoined.substr(0, ALGLIST[algName]['keylen'] * 2);
205 	result.ivhex = hHexValueJoined.substr(ALGLIST[algName]['keylen'] * 2, ALGLIST[algName]['ivlen'] * 2);
206 	return result;
207     };

374          * get PEM formatted encrypted PKCS#5 private key from hexadecimal string of plain private key
375 	 * @name getEncryptedPKCS5PEMFromPrvKeyHex
376 	 * @memberOf PKCS5PKEY
377 	 * @function
378 	 * @param {String} hPrvKey hexadecimal string of plain private key
379 	 * @param {String} passcode pass code to protect private key (ex. password)
380 	 * @param {String} sharedKeyAlgName algorithm name to protect private key (ex. AES-256-CBC)
381 	 * @param {String} ivsaltHex hexadecimal string of IV and salt
382 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
383          * @since pkcs5pkey 1.0.2
384 	 * @description
385 	 * <br/>
386 	 * generate PEM formatted encrypted PKCS#5 private key by hexadecimal string encoded
387 	 * ASN.1 object of plain RSA private key.
388 	 * Following arguments can be omitted.
389 	 * <ul>
390 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
391 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
392 	 * </ul>
393 	 * @example
394 	 * var pem = 
395          *   PKCS5PKEY.getEncryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password");
396 	 * var pem2 = 
397          *   PKCS5PKEY.getEncryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC");
398 	 * var pem3 = 
399          *   PKCS5PKEY.getEncryptedPKCS5PEMFromPrvKeyHex(plainKeyHex, "password", "AES-128-CBC", "1f3d02...");
400 	 */
401 	getEncryptedPKCS5PEMFromPrvKeyHex: function(hPrvKey, passcode, sharedKeyAlgName, ivsaltHex) {
402 	    var sPEM = "";
403 
404 	    // 1. set sharedKeyAlgName if undefined (default AES-256-CBC)
405 	    if (typeof sharedKeyAlgName == "undefined" || sharedKeyAlgName == null) {
406 		sharedKeyAlgName = "AES-256-CBC";
407 	    }
408 	    if (typeof ALGLIST[sharedKeyAlgName] == "undefined")
409 		throw "PKCS5PKEY unsupported algorithm: " + sharedKeyAlgName;
410 
411 	    // 2. set ivsaltHex if undefined
412 	    if (typeof ivsaltHex == "undefined" || ivsaltHex == null) {
413 		var ivlen = ALGLIST[sharedKeyAlgName]['ivlen'];
414 		var randIV = _generateIvSaltHex(ivlen);
415 		ivsaltHex = randIV.toUpperCase();
416 	    }
417 
418 	    // 3. get shared key
419             //alert("ivsalthex=" + ivsaltHex);
420 	    var sharedKeyInfo = _getKeyAndUnusedIvByPasscodeAndIvsalt(sharedKeyAlgName, passcode, ivsaltHex);
421 	    var sharedKeyHex = sharedKeyInfo.keyhex;
422 	    // alert("sharedKeyHex = " + sharedKeyHex);
423 
424             // 3. get encrypted Key in Base64
425             var encryptedKeyB64 = _encryptKeyHex(hPrvKey, sharedKeyAlgName, sharedKeyHex, ivsaltHex);
426 
427 	    var pemBody = encryptedKeyB64.replace(/(.{64})/g, "$1\r\n");
428 	    var sPEM = "-----BEGIN RSA PRIVATE KEY-----\r\n";
429 	    sPEM += "Proc-Type: 4,ENCRYPTED\r\n";
430 	    sPEM += "DEK-Info: " + sharedKeyAlgName + "," + ivsaltHex + "\r\n";
431 	    sPEM += "\r\n";
432 	    sPEM += pemBody;
433 	    sPEM += "\r\n-----END RSA PRIVATE KEY-----\r\n";
434 
435 	    return sPEM;
436         },

438 	/**
439          * get PEM formatted encrypted PKCS#5 private key from RSAKey object of private key
440 	 * @name getEncryptedPKCS5PEMFromRSAKey
441 	 * @memberOf PKCS5PKEY
442 	 * @function
443 	 * @param {RSAKey} pKey RSAKey object of private key
444 	 * @param {String} passcode pass code to protect private key (ex. password)
445 	 * @param {String} alg algorithm name to protect private key (default AES-256-CBC)
446 	 * @param {String} ivsaltHex hexadecimal string of IV and salt (default generated random IV)
447 	 * @return {String} string of PEM formatted encrypted PKCS#5 private key
448          * @since pkcs5pkey 1.0.2
449 	 * @description
450 	 * <br/>
451 	 * generate PEM formatted encrypted PKCS#5 private key by
452 	 * {@link RSAKey} object of RSA private key and passcode.
453 	 * Following argument can be omitted.
454 	 * <ul>
455 	 * <li>alg - AES-256-CBC will be used if omitted.</li>
456 	 * <li>ivsaltHex - automatically generate IV and salt which length depends on algorithm</li>
457 	 * </ul>
458 	 * @example
459 	 * var pkey = new RSAKey();
460 	 * pkey.generate(1024, '10001'); // generate 1024bit RSA private key with public exponent 'x010001'
461 	 * var pem = PKCS5PKEY.getEncryptedPKCS5PEMFromRSAKey(pkey, "password");
462 	 */
463         getEncryptedPKCS5PEMFromRSAKey: function(pKey, passcode, alg, ivsaltHex) {
464 	    var version = new KJUR.asn1.DERInteger({'int': 0});
465 	    var n = new KJUR.asn1.DERInteger({'bigint': pKey.n});
466 	    var e = new KJUR.asn1.DERInteger({'int': pKey.e});
467 	    var d = new KJUR.asn1.DERInteger({'bigint': pKey.d});
468 	    var p = new KJUR.asn1.DERInteger({'bigint': pKey.p});
469 	    var q = new KJUR.asn1.DERInteger({'bigint': pKey.q});
470 	    var dmp1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmp1});
471 	    var dmq1 = new KJUR.asn1.DERInteger({'bigint': pKey.dmq1});
472 	    var coeff = new KJUR.asn1.DERInteger({'bigint': pKey.coeff});
473 	    var seq = new KJUR.asn1.DERSequence({'array': [version, n, e, d, p, q, dmp1, dmq1, coeff]});
474 	    var hex = seq.getEncodedHex();
475 	    return this.getEncryptedPKCS5PEMFromPrvKeyHex(hex, passcode, alg, ivsaltHex);
476         },
